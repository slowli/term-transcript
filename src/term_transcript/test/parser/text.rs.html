50impl TextReadingState {

63    // We only retain `<span>` tags in the HTML since they are the only ones containing color info.
64    #[allow(clippy::too_many_lines)]
65    pub(super) fn process(
66        &mut self,
67        event: Event<'_>,
68        position: ops::Range<usize>,
69    ) -> Result<Option<Parsed>, ParseError> {
70        let after_hard_break = matches!(self.hard_br, Some(HardBreak::JustEnded));
71        if after_hard_break
72            && matches!(
73                &event,
74                Event::Text(_) | Event::GeneralRef(_) | Event::Start(_)
75            )
76        {
77            self.hard_br = None;
78        }
79
80        match event {
81            Event::Text(text) => {
82                if self.should_ignore_text() {
83                    return Ok(None);
84                }
85
86                let unescaped_str = text.decode().map_err(quick_xml::Error::from)?;
87                let unescaped_str = normalize_newlines(&unescaped_str);
88                let unescaped_str = if after_hard_break && unescaped_str.starts_with('\n') {
89                    &unescaped_str[1..] // gobble the starting '\n' as produced by a hard break
90                } else {
91                    &unescaped_str
92                };
93                self.push_text(unescaped_str);
94            }
95            Event::GeneralRef(reference) => {
96                if self.should_ignore_text() {
97                    return Ok(None);
98                }
99
100                let maybe_char = reference.resolve_char_ref()?;
101                let mut char_buffer = [0_u8; 4];
102                let decoded = if let Some(c) = maybe_char {
103                    c.encode_utf8(&mut char_buffer)
104                } else {
105                    let decoded = reference.decode().map_err(quick_xml::Error::from)?;
106                    resolve_xml_entity(&decoded).ok_or_else(|| {
107                        let err = EscapeError::UnrecognizedEntity(position, decoded.into_owned());
108                        quick_xml::Error::from(err)
109                    })?
110                };
111                self.push_text(decoded);
112            }
113            Event::Start(tag) => {
114                self.open_tags += 1;
115                if self.hard_br.is_some() {
116                    return Err(ParseError::InvalidHardBreak);
117                }
118
119                let tag_name = tag.name();
120                // Check for the hard line break <text> or <b>. We mustn't add its contents to the text,
121                // and instead gobble the following '\n'.
122                let classes = parse_classes(tag.attributes())?;
123                if extract_base_class(&classes) == b"hard-br" {
124                    self.hard_br = Some(HardBreak::Active);
125                    return Ok(None);
126                }
127
128                if Self::is_text_span(tag_name.as_ref()) {
129                    let style = Self::parse_style_from_span(&tag)?;
130                    if !style.is_none() {
131                        self.color_spans_writer
132                            .write_style(&style)
133                            .expect("cannot set color for ANSI buffer");
134                    }
135                }
136            }
137            Event::End(tag) => {
138                self.open_tags -= 1;
139                if matches!(self.hard_br, Some(HardBreak::Active)) {
140                    self.hard_br = Some(HardBreak::JustEnded);
141                    return Ok(None);
142                }
143
144                if Self::is_text_span(tag.name().as_ref()) {
145                    self.color_spans_writer
146                        .write_style(&Style::default())
147                        .expect("cannot reset color for ANSI buffer");
148                }
149
150                if self.open_tags == 0 {
151                    let plaintext = mem::take(&mut self.plaintext_buffer);
152                    let color_spans = mem::take(&mut self.color_spans_writer).into_inner();
153                    let mut parsed = Parsed {
154                        plaintext,
155                        styled_spans: color_spans,
156                    };
157                    parsed.trim_ending_newline();
158                    return Ok(Some(parsed));
159                }
160            }
161            _ => { /* Do nothing */ }
162        }
163        Ok(None)
164    }
165
166    fn is_text_span(tag: &[u8]) -> bool {
167        matches!(tag, b"span" | b"tspan" | b"text")
168    }
169
170    fn push_text(&mut self, text: &str) {
171        self.plaintext_buffer.push_str(text);
172        self.color_spans_writer
173            .write_text(text)
174            .expect("cannot write to ANSI buffer");
175    }

177    /// Parses a style from a `span`.
178    ///
179    /// **NB.** Must correspond to the span creation logic in the `svg` module.
180    fn parse_style_from_span(span_tag: &BytesStart) -> Result<Style, ParseError> {
181        let class_attr = parse_classes(span_tag.attributes())?;
182        let mut style = Style::default();
183        Self::parse_color_from_classes(&mut style, &class_attr);
184
185        let mut style_attr = Cow::Borrowed(&[] as &[u8]);
186        for attr in span_tag.attributes() {
187            let attr = attr.map_err(quick_xml::Error::InvalidAttr)?;
188            if attr.key.as_ref() == b"style" {
189                style_attr = attr.value;
190            }
191        }
192        Self::parse_color_from_style(&mut style, &style_attr)?;
193
194        if style.inverted {
195            // Swap fg and bg colors back; they are swapped when writing to SVG.
196            mem::swap(&mut style.fg, &mut style.bg);
197        }
198
199        Ok(style)
200    }
201
202    fn parse_color_from_classes(style: &mut Style, class_attr: &[u8]) {
203        let classes = class_attr.split(u8::is_ascii_whitespace);
204        for class in classes {
205            // Note that `class` may be empty because of multiple sequential whitespace chars.
206            // This is OK for us.
207            match class {
208                b"bold" => {
209                    style.bold = true;
210                }
211                b"dimmed" => {
212                    style.dimmed = true;
213                }
214                b"italic" => {
215                    style.italic = true;
216                }
217                b"underline" => {
218                    style.underline = true;
219                }
220                b"strike" => {
221                    style.strikethrough = true;
222                }
223                b"blink" => {
224                    style.blink = true;
225                }
226                b"concealed" => {
227                    style.concealed = true;
228                }
229                b"inv" => {
230                    style.inverted = true;
231                }
232
233                // Indexed foreground color candidate.
234                fg if fg.starts_with(b"fg") => {
235                    if let Some(color) = Self::parse_indexed_color(&fg[2..]) {
236                        style.fg = Some(color);
237                    }
238                }
239                // Indexed background color candidate.
240                bg if bg.starts_with(b"bg") => {
241                    if let Some(color) = Self::parse_indexed_color(&bg[2..]) {
242                        style.bg = Some(color);
243                    } else if let Ok(color_str) = str::from_utf8(&bg[2..]) {
244                        // Parse `bg#..` classes produced by the pure SVG template
245                        if let Ok(color) = color_str.parse::<RgbColor>() {
246                            style.bg = Some(Color::Rgb(color));
247                        }
248                    }
249                }
250
251                _ => { /* Ignore other classes. */ }
252            }
253        }
254    }

256    // **NB.** This parser is pretty rudimentary (e.g., does not understand comments).
257    fn parse_color_from_style(color_spec: &mut Style, style: &[u8]) -> Result<(), ParseError> {
258        for style_property in style.split(|&ch| ch == b';') {
259            let name_and_value: Vec<_> = style_property.splitn(2, |&ch| ch == b':').collect();
260            let [property_name, property_value] = name_and_value.as_slice() else {
261                continue;
262            };
263
264            let property_name = str::from_utf8(property_name)
265                .map_err(map_utf8_error)?
266                .trim();
267            let property_value = str::from_utf8(property_value)
268                .map_err(map_utf8_error)?
269                .trim();
270
271            match property_name {
272                "color" | "fill" => {
273                    if let Ok(color) = property_value.parse::<RgbColor>() {
274                        color_spec.fg = Some(Color::Rgb(color));
275                    }
276                }
277                "background" | "background-color" => {
278                    if let Ok(color) = property_value.parse::<RgbColor>() {
279                        color_spec.bg = Some(Color::Rgb(color));
280                    }
281                }
282                _ => { /* Ignore other properties. */ }
283            }
284        }
285        Ok(())
286    }
287
288    fn parse_indexed_color(class: &[u8]) -> Option<Color> {
289        Some(match class {
290            b"0" => Color::BLACK,
291            b"1" => Color::RED,
292            b"2" => Color::GREEN,
293            b"3" => Color::YELLOW,
294            b"4" => Color::BLUE,
295            b"5" => Color::MAGENTA,
296            b"6" => Color::CYAN,
297            b"7" => Color::WHITE,
298            b"8" | b"9" => Color::Index(class[0] - b'0'),
299            b"10" | b"11" | b"12" | b"13" | b"14" | b"15" => Color::Index(10 + class[1] - b'0'),
300            _ => return None,
301        })
302    }
