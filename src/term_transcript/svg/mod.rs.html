80/// Configurable options of a [`Template`].
81///
82/// # Serialization
83///
84/// Options can be deserialized from `serde`-supported encoding formats, such as TOML. This is used
85/// in the CLI app to read options from a file:
86///
87/// ```
88/// # use assert_matches::assert_matches;
89/// # use term_transcript::svg::{RgbColor, TemplateOptions, WrapOptions};
90/// let options_toml = r#"
91/// width = 900
92/// window_frame = true
93/// line_numbers = 'continuous'
94/// wrap.hard_break_at = 100
95/// scroll = { max_height = 300, pixels_per_scroll = 18, interval = 1.5 }
96///
97/// [palette.colors]
98/// black = '#3c3836'
99/// red = '#b85651'
100/// green = '#8f9a52'
101/// yellow = '#c18f41'
102/// blue = '#68948a'
103/// magenta = '#ab6c7d'
104/// cyan = '#72966c'
105/// white = '#a89984'
106///
107/// [palette.intense_colors]
108/// black = '#5a524c'
109/// red = '#b85651'
110/// green = '#a9b665'
111/// yellow = '#d8a657'
112/// blue = '#7daea3'
113/// magenta = '#d3869b'
114/// cyan = '#89b482'
115/// white = '#ddc7a1'
116/// "#;
117///
118/// let options: TemplateOptions = toml::from_str(options_toml)?;
119/// assert_eq!(options.width.get(), 900);
120/// assert_matches!(
121///     options.wrap,
122///     Some(WrapOptions::HardBreakAt(width)) if width.get() == 100
123/// );
124/// assert_eq!(
125///     options.palette.colors.green,
126///     RgbColor(0x8f, 0x9a, 0x52)
127/// );
128/// # anyhow::Ok(())
129/// ```
130#[derive(Debug, Serialize, Deserialize)]
131pub struct TemplateOptions {
132    /// Width of the rendered terminal window in pixels. Excludes the line numbers width if line
133    /// numbering is enabled. The default value is `720`.
134    #[serde(default = "TemplateOptions::default_width")]
135    pub width: NonZeroUsize,
136    /// Line height relative to the font size. If not specified, will be taken from font metrics (if a font is embedded),
137    /// or set to 1.2 otherwise.
138    pub line_height: Option<f64>,
139    /// Advance width of a font relative to the font size (i.e., in em units). If not specified, will be taken from font metrics (if a font is embedded),
140    /// or set to 8px (~0.57em) otherwise.
141    ///
142    /// For now, advance width is only applied to the pure SVG template.
143    // FIXME: extract to pure SVG options?
144    pub advance_width: Option<f64>,
145    /// Palette of terminal colors. The default value of [`Palette`] is used by default.
146    #[serde(default)]
147    pub palette: Palette,
148    /// CSS instructions to add at the beginning of the SVG `<style>` tag. This is mostly useful
149    /// to import fonts in conjunction with `font_family`.
150    ///
151    /// The value is not validated in any way, so supplying invalid CSS instructions can lead
152    /// to broken SVG rendering.
153    #[serde(skip_serializing_if = "str::is_empty", default)]
154    pub additional_styles: String,
155    /// Font family specification in the CSS format. Should be monospace.
156    #[serde(default = "TemplateOptions::default_font_family")]
157    pub font_family: String,
158    /// Indicates whether to display a window frame around the shell. Default value is `false`.
159    #[serde(default)]
160    pub window_frame: bool,
161    /// Options for the scroll animation. If set to `None` (which is the default),
162    /// no scrolling will be enabled, and the height of the generated image is not limited.
163    #[serde(skip_serializing_if = "Option::is_none", default)]
164    pub scroll: Option<ScrollOptions>,
165    /// Text wrapping options. The default value of [`WrapOptions`] is used by default.
166    #[serde(default = "TemplateOptions::default_wrap")]
167    pub wrap: Option<WrapOptions>,
168    /// Line numbering options.
169    #[serde(default)]
170    pub line_numbers: Option<LineNumbers>,
171    /// *Font embedder* that will embed the font into the SVG file via `@font-face` CSS.
172    /// This guarantees that the SVG will look identical on all platforms.
173    #[serde(skip)]
174    pub font_embedder: Option<Box<dyn FontEmbedder<Error = BoxedError>>>,
175}

195impl TemplateOptions {
196    fn validate(&self) -> anyhow::Result<()> {
197        if let Some(line_height) = self.line_height {
198            anyhow::ensure!(line_height > 0.0, "line_height must be positive");
199            #[cfg(feature = "tracing")]
200            if line_height > 2.0 {
201                tracing::warn!(
202                    line_height,
203                    "line_height is too large, the produced SVG may look broken"
204                );
205            }
206        }
207
208        if let Some(advance_width) = self.advance_width {
209            anyhow::ensure!(advance_width > 0.0, "advance_width must be positive");
210            #[cfg(feature = "tracing")]
211            if advance_width > 0.7 {
212                tracing::warn!(
213                    advance_width,
214                    "advance_width is too large, the produced SVG may look broken"
215                );
216            }
217            #[cfg(feature = "tracing")]
218            if advance_width < 0.5 {
219                tracing::warn!(
220                    advance_width,
221                    "advance_width is too small, the produced SVG may look broken"
222                );
223            }
224        }
225
226        if let Some(scroll_options) = &self.scroll {
227            scroll_options
228                .validate()
229                .context("invalid scroll options")?;
230        }
231
232        Ok(())
233    }

271    #[cfg_attr(
272        feature = "tracing",
273        tracing::instrument(level = "debug", skip(transcript), err)
274    )]
275    fn render_data<'s>(
276        &'s self,
277        transcript: &'s Transcript,
278    ) -> Result<HandlebarsData<'s>, TermError> {
279        let rendered_outputs = self.render_outputs(transcript)?;
280        let mut has_failures = false;
281
282        let mut used_chars = BTreeSet::new();
283        for interaction in transcript.interactions() {
284            let output = interaction.output().to_plaintext()?;
285            used_chars.extend(output.chars());
286
287            let input = interaction.input();
288            if !input.hidden {
289                let prompt = input.prompt.as_deref();
290                let input_chars = iter::once(input.text.as_str())
291                    .chain(prompt)
292                    .flat_map(str::chars);
293                used_chars.extend(input_chars);
294            }
295        }
296        if self.line_numbers.is_some() {
297            used_chars.extend('0'..='9');
298        }
299        if self.wrap.is_some() {
300            used_chars.insert('»');
301        }
302
303        let embedded_font = self
304            .font_embedder
305            .as_deref()
306            .map(|embedder| embedder.embed_font(used_chars))
307            .transpose()
308            .map_err(TermError::FontEmbedding)?;
309
310        let interactions: Vec<_> = transcript
311            .interactions()
312            .iter()
313            .zip(rendered_outputs)
314            .map(|(interaction, output)| {
315                let failure = interaction
316                    .exit_status()
317                    .is_some_and(|status| !status.is_success());
318                has_failures = has_failures || failure;
319                SerializedInteraction {
320                    input: interaction.input(),
321                    output,
322                    exit_status: interaction.exit_status().map(|status| status.0),
323                    failure,
324                }
325            })
326            .collect();
327
328        Ok(HandlebarsData {
329            creator: CreatorData::default(),
330            interactions,
331            options: self,
332            has_failures,
333            embedded_font,
334        })
335    }
336
337    #[cfg_attr(
338        feature = "tracing",
339        tracing::instrument(level = "debug", skip_all, err)
340    )]
341    fn render_outputs(&self, transcript: &Transcript) -> Result<Vec<Vec<StyledLine>>, TermError> {
342        let max_width = self.wrap.as_ref().map(|wrap_options| match wrap_options {
343            WrapOptions::HardBreakAt(width) => width.get(),
344        });
345
346        transcript
347            .interactions
348            .iter()
349            .map(|interaction| interaction.output().to_lines(max_width))
350            .collect()
351    }

354/// Options that influence the scrolling animation.
355///
356/// The animation is only displayed if the console exceeds [`Self::max_height`]. In this case,
357/// the console will be scrolled vertically by [`Self::pixels_per_scroll`]
358/// with the interval of [`Self::interval`] seconds between every frame.
359#[derive(Debug, Clone, Deserialize, Serialize)]
360#[cfg_attr(test, derive(PartialEq))]
361pub struct ScrollOptions {
362    /// Maximum height of the console, in pixels. The default value allows to fit 19 lines
363    /// of text into the view with the default template (potentially, slightly less because
364    /// of vertical margins around user inputs).
365    #[serde(default = "ScrollOptions::default_max_height")]
366    pub max_height: NonZeroUsize,
367    /// Minimum scrollbar height in pixels. The default value is 14px (1em).
368    #[serde(default = "ScrollOptions::default_min_scrollbar_height")]
369    pub min_scrollbar_height: NonZeroUsize,
370    /// Number of pixels moved each scroll. Default value is 52 (~3 lines of text with the default template).
371    #[serde(default = "ScrollOptions::default_pixels_per_scroll")]
372    pub pixels_per_scroll: NonZeroUsize,
373    /// Interval between keyframes in seconds. The default value is `4`.
374    #[serde(default = "ScrollOptions::default_interval")]
375    pub interval: f64,
376    /// Threshold to elide the penultimate scroll keyframe, relative to `pixels_per_scroll`.
377    /// If the last scroll keyframe would scroll the view by less than this value (which can happen because
378    /// the last scroll always aligns the scrolled view bottom with the viewport bottom), it will be
379    /// combined with the penultimate keyframe.
380    ///
381    /// The threshold must be in [0, 1). 0 means never eliding the penultimate keyframe. The default value is 0.25.
382    #[serde(default = "ScrollOptions::default_elision_threshold")]
383    pub elision_threshold: f64,
384}
385
386impl Default for ScrollOptions {
387    fn default() -> Self {
388        Self::DEFAULT
389    }
390}

396        min_scrollbar_height: Self::default_min_scrollbar_height(),
397        pixels_per_scroll: Self::default_pixels_per_scroll(),
398        interval: Self::default_interval(),
399        elision_threshold: Self::default_elision_threshold(),
400    };

402    const fn default_max_height() -> NonZeroUsize {
403        NonZeroUsize::new(18 * 19).unwrap()
404    }
405
406    const fn default_min_scrollbar_height() -> NonZeroUsize {
407        NonZeroUsize::new(14).unwrap()
408    }
409
410    const fn default_pixels_per_scroll() -> NonZeroUsize {
411        NonZeroUsize::new(52).unwrap()
412    }
413
414    const fn default_interval() -> f64 {
415        4.0
416    }
417
418    const fn default_elision_threshold() -> f64 {
419        0.25
420    }
421
422    fn validate(&self) -> anyhow::Result<()> {
423        anyhow::ensure!(self.interval > 0.0, "interval must be positive");
424        anyhow::ensure!(
425            self.elision_threshold >= 0.0 && self.elision_threshold < 1.0,
426            "elision_threshold must be in [0, 1)"
427        );
428
429        anyhow::ensure!(
430            self.min_scrollbar_height < self.max_height,
431            "min_scrollbar_height={} must be lesser than max_height={}",
432            self.min_scrollbar_height,
433            self.max_height
434        );
435        Ok(())
436    }

497/// Template for rendering [`Transcript`]s, e.g. into an [SVG] image.
498///
499/// # Available templates
500///
501/// When using a template created with [`Self::new()`], a transcript is rendered into SVG
502/// with the text content embedded as an HTML fragment. This is because SVG is not good
503/// at laying out multiline texts and text backgrounds, while HTML excels at both.
504/// As a downside of this approach, the resulting SVG requires for its viewer to support
505/// HTML embedding; while web browsers *a priori* support such embedding, some other SVG viewers
506/// may not.
507///
508/// A template created with [`Self::pure_svg()`] renders a transcript into pure SVG,
509/// in which text is laid out manually and backgrounds use a hack (lines of text with
510/// appropriately colored `█` chars placed behind the content lines). The resulting SVG is
511/// supported by more viewers, but it may look incorrectly in certain corner cases. For example,
512/// if the font family used in the template does not contain `█` or some chars
513/// used in the transcript, the background may be mispositioned.
514///
515/// [Snapshot testing](crate::test) functionality produces snapshots using [`Self::new()`]
516/// (i.e., with HTML embedding); pure SVG templates cannot be tested.
517///
518/// # Customization
519///
520/// A custom [Handlebars] template can be supplied via [`Self::custom()`]. This can be used
521/// to partially or completely change rendering logic, including the output format (e.g.,
522/// to render to HTML instead of SVG).
523///
524/// Data supplied to a template is [`HandlebarsData`].
525///
526/// Besides [built-in Handlebars helpers][rust-helpers] (a superset of [standard helpers]),
527/// custom templates have access to the following additional helpers. All the helpers are
528/// extensively used by the [default template]; thus, studying it may be a good place to start
529/// customizing. Another example is an [HTML template] from the crate examples.
530///
531/// ## Arithmetic helpers: `add`, `sub`, `mul`, `div`
532///
533/// Perform the specified arithmetic operation on the supplied args.
534/// `add` and `mul` support any number of numeric args; `sub` and `div` exactly 2 numeric args.
535/// `div` also supports rounding via `round` hash option. `round=true` rounds to the nearest
536/// integer; `round="up"` / `round="down"` perform rounding in the specified direction.
537///
538/// ```handlebars
539/// {{add 2 3 5}}
540/// {{div (len xs) 3 round="up"}}
541/// ```
542///
543/// ## Counting lines: `count_lines`
544///
545/// Counts the number of lines in the supplied string. If `format="html"` hash option is included,
546/// line breaks introduced by `<br/>` tags are also counted.
547///
548/// ```handlebars
549/// {{count_lines test}}
550/// {{count_lines test format="html"}}
551/// ```
552///
553/// ## Integer ranges: `range`
554///
555/// Creates an array with integers in the range specified by the 2 provided integer args.
556/// The "from" bound is inclusive, the "to" one is exclusive.
557///
558/// ```handlebars
559/// {{#each (range 0 3)}}{{@index}}, {{/each}}
560/// {{! Will output `0, 1, 2,` }}
561/// ```
562///
563/// ## Variable scope: `scope`
564///
565/// A block helper that creates a scope with variables specified in the options hash.
566/// In the block, each variable can be obtained or set using an eponymous helper:
567///
568/// - If the variable helper is called as a block helper, the variable is set to the contents
569///   of the block, which is treated as JSON.
570/// - If the variable helper is called as an inline helper with the `set` option, the variable
571///   is set to the value of the option.
572/// - Otherwise, the variable helper acts as a getter for the current value of the variable.
573///
574/// ```handlebars
575/// {{#scope test=""}}
576///   {{test set="Hello"}}
577///   {{test}} {{! Outputs `Hello` }}
578///   {{#test}}{{test}}, world!{{/test}}
579///   {{test}} {{! Outputs `Hello, world!` }}
580/// {{/scope}}
581/// ```
582///
583/// Since variable getters are helpers, not "real" variables, they should be enclosed
584/// in parentheses `()` if used as args / options for other helpers, e.g. `{{add (test) 2}}`.
585///
586/// ## Partial evaluation: `eval`
587///
588/// Evaluates a partial with the provided name and parses its output as JSON. This can be used
589/// to define "functions" for better code structuring. Function args can be supplied in options
590/// hash.
591///
592/// ```handlebars
593/// {{#*inline "some_function"}}
594///   {{add x y}}
595/// {{/inline}}
596/// {{#with (eval "some_function" x=3 y=5) as |sum|}}
597///   {{sum}} {{! Outputs 8 }}
598/// {{/with}}
599/// ```
600///
601/// [SVG]: https://developer.mozilla.org/en-US/docs/Web/SVG
602/// [Handlebars]: https://handlebarsjs.com/
603/// [rust-helpers]: https://docs.rs/handlebars/latest/handlebars/index.html#built-in-helpers
604/// [standard helpers]: https://handlebarsjs.com/guide/builtin-helpers.html
605/// [default template]: https://github.com/slowli/term-transcript/blob/master/src/svg/default.svg.handlebars
606/// [HTML template]: https://github.com/slowli/term-transcript/blob/master/examples/custom.html.handlebars
607///
608/// # Examples
609///
610/// ```
611/// use term_transcript::{svg::*, Transcript, UserInput};
612///
613/// let mut transcript = Transcript::new();
614/// transcript.add_interaction(
615///     UserInput::command("test"),
616///     "Hello, \u{1b}[32mworld\u{1b}[0m!",
617/// );
618///
619/// let template_options = TemplateOptions {
620///     palette: NamedPalette::Dracula.into(),
621///     ..TemplateOptions::default()
622/// }
623/// .validated()?;
624/// let mut buffer = vec![];
625/// Template::new(template_options).render(&transcript, &mut buffer)?;
626/// let buffer = String::from_utf8(buffer)?;
627/// assert!(buffer.contains(r#"Hello, <span class="fg2">world</span>!"#));
628/// # anyhow::Ok(())
629/// ```
630pub struct Template {
631    options: TemplateOptions,
632    handlebars: Handlebars<'static>,
633    constants: HashMap<&'static str, u32>,
634}

664    /// Initializes the default template based on provided `options`.
665    #[allow(clippy::missing_panics_doc)] // Panic should never be triggered
666    pub fn new(options: ValidTemplateOptions) -> Self {
667        let template = HandlebarsTemplate::compile(DEFAULT_TEMPLATE)
668            .expect("Default template should be valid");
669        Self {
670            constants: Self::STD_CONSTANTS.iter().copied().collect(),

672        }

686    /// Initializes a custom template.
687    #[allow(clippy::missing_panics_doc)] // Panic should never be triggered
688    pub fn custom(template: HandlebarsTemplate, options: ValidTemplateOptions) -> Self {
689        let mut handlebars = Handlebars::new();
690        handlebars.set_strict_mode(true);
691        register_helpers(&mut handlebars);
692        handlebars.register_template(MAIN_TEMPLATE_NAME, template);
693        let helpers = HandlebarsTemplate::compile(COMMON_HELPERS).unwrap();
694        handlebars.register_template("_helpers", helpers);
695        Self {
696            options: options.0,
697            handlebars,
698            constants: HashMap::new(),
699        }
700    }

702    /// Renders the `transcript` using the template (usually as an SVG image, although
703    /// custom templates may use a different output format).
704    ///
705    /// # Errors
706    ///
707    /// Returns a Handlebars rendering error, if any. Normally, the only errors could be
708    /// related to I/O (e.g., the output cannot be written to a file).
709    #[cfg_attr(feature = "tracing", tracing::instrument(skip_all, err))]
710    pub fn render<W: Write>(
711        &self,
712        transcript: &Transcript,
713        destination: W,
714    ) -> Result<(), RenderError> {
715        let data = self
716            .options
717            .render_data(transcript)
718            .map_err(|err| RenderErrorReason::NestedError(Box::new(err)))?;
719        let data = CompleteHandlebarsData {
720            inner: data,
721            constants: &self.constants,
722        };
723        #[cfg(feature = "tracing")]
724        tracing::debug!(?data, "using Handlebars data");
725
726        #[cfg(feature = "tracing")]
727        let _entered = tracing::debug_span!("render_to_write").entered();
728        self.handlebars
729            .render_to_write(MAIN_TEMPLATE_NAME, &data, destination)
730    }
