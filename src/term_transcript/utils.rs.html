3use std::{borrow::Cow, io, str};
4
5pub(crate) fn normalize_newlines(s: &str) -> Cow<'_, str> {
6    if s.contains("\r\n") {
7        Cow::Owned(s.replace("\r\n", "\n"))
8    } else {
9        Cow::Borrowed(s)
10    }
11}

13#[cfg(not(windows))]
14pub(crate) fn is_recoverable_kill_error(err: &io::Error) -> bool {
15    matches!(err.kind(), io::ErrorKind::InvalidInput)
16}

18// As per `TerminateProcess` docs (`TerminateProcess` is used by `Child::kill()`),
19// the call will result in ERROR_ACCESS_DENIED if the process has already terminated.
20//
21// https://docs.microsoft.com/en-us/windows/win32/api/processthreadsapi/nf-processthreadsapi-terminateprocess
22#[cfg(windows)]
23pub(crate) fn is_recoverable_kill_error(err: &io::Error) -> bool {
24    matches!(
25        err.kind(),
26        io::ErrorKind::InvalidInput | io::ErrorKind::PermissionDenied
27    )
28}
