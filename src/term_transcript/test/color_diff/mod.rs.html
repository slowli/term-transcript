20impl StyledSpan<usize> {
21    pub(crate) fn parse(ansi_text: &str) -> Result<Vec<Self>, TermError> {
22        let mut spans = ColorSpansWriter::default();
23        TermOutputParser::new(&mut spans).parse(ansi_text.as_bytes())?;
24        Ok(spans.shrink().spans)
25    }
26
27    pub(crate) fn write_colorized(
28        spans: &[Self],
29        out: &mut impl Write,
30        plaintext: &str,
31    ) -> io::Result<()> {
32        debug_assert_eq!(
33            spans.iter().map(|span| span.text).sum::<usize>(),
34            plaintext.len()
35        );
36        let mut pos = 0;
37        for span in spans {
38            write!(
39                out,
40                "{style}{}{style:#}",
41                &plaintext[pos..pos + span.text],
42                style = span.style
43            )?;
44            pos += span.text;
45        }
46        Ok(())
47    }
48
49    /// Writes a single plaintext `line` to `out` using styles from `spans_iter`.
50    fn write_line<'a, I>(
51        spans_iter: &mut Peekable<I>,
52        out: &mut impl Write,
53        line_start: usize,
54        line: &str,
55    ) -> io::Result<()>
56    where
57        I: Iterator<Item = (usize, &'a Self)>,
58    {
59        let mut pos = 0;
60        while pos < line.len() {
61            let &(span_start, span) = spans_iter.peek().expect("spans ended before lines");
62            let span_len = span.text - line_start.saturating_sub(span_start);
63
64            let span_end = cmp::min(pos + span_len, line.len());
65            write!(
66                out,
67                "{style}{}{style:#}",
68                &line[pos..span_end],
69                style = span.style
70            )?;
71            if span_end == pos + span_len {
72                // The span has ended, can proceed to the next one.
73                spans_iter.next();
74            }
75            pos += span_len;
76        }
77        writeln!(out)
78    }
79}

81/// `Write` / `WriteColor` implementation recording `ColorSpan`s for the input text.
82#[derive(Debug, Default)]
83pub(crate) struct ColorSpansWriter {
84    spans: Vec<StyledSpan<usize>>,
85    style: Style,
86}
87
88impl ColorSpansWriter {
89    /// Unites sequential spans with the same color spec.
90    fn shrink(self) -> Self {
91        let mut shrunk_spans = Vec::<StyledSpan<_>>::with_capacity(self.spans.len());
92        for span in self.spans {
93            if let Some(last_span) = shrunk_spans.last_mut() {
94                if last_span.style == span.style {
95                    last_span.text += span.text;
96                } else {
97                    shrunk_spans.push(span);
98                }
99            } else {
100                shrunk_spans.push(span);
101            }
102        }
103
104        Self {
105            spans: shrunk_spans,
106            style: self.style,
107        }
108    }
109
110    pub(crate) fn into_inner(self) -> Vec<StyledSpan<usize>> {
111        self.shrink().spans
112    }
113}

115impl WriteStyled for ColorSpansWriter {
116    fn write_style(&mut self, style: &Style) -> io::Result<()> {
117        let mut normalized_style = *style;
118        normalized_style.normalize();
119        self.style = normalized_style;
120        Ok(())
121    }
122
123    fn write_text(&mut self, text: &str) -> io::Result<()> {
124        // Break styling on newlines because it will be broken in the parsed transcripts.
125        let lines = text.split('\n');
126        let mut pos = 0;
127        self.spans.extend(lines.flat_map(|line| {
128            let mut new_spans = vec![];
129            if !line.is_empty() {
130                new_spans.push(StyledSpan {
131                    style: self.style,
132                    text: line.len(),
133                });
134            }
135            pos += line.len();
136            if pos < text.len() {
137                new_spans.push(StyledSpan {
138                    style: Style::default(),
139                    text: 1,
140                });
141                pos += 1;
142            }
143            new_spans
144        }));
145        Ok(())
146    }
147}

150struct DiffColorSpan {
151    start: usize,
152    len: usize,
153    lhs_color_spec: Style,
154    rhs_color_spec: Style,
155}

159#[derive(Debug, Default)]
160pub(crate) struct ColorDiff {
161    differing_spans: Vec<DiffColorSpan>,
162}

164impl ColorDiff {
165    pub(crate) fn new(lhs: &[StyledSpan<usize>], rhs: &[StyledSpan<usize>]) -> Self {
166        debug_assert_eq!(
167            lhs.iter().map(|span| span.text).sum::<usize>(),
168            rhs.iter().map(|span| span.text).sum::<usize>(),
169            "Spans {lhs:?} and {rhs:?} must have equal total covered length"
170        );
171
172        let mut diff = Self::default();
173        let mut pos = 0;
174        let mut lhs_iter = lhs.iter().copied();
175        let Some(mut lhs_span) = lhs_iter.next() else {
176            return diff;
177        };
178        let mut rhs_iter = rhs.iter().copied();
179        let Some(mut rhs_span) = rhs_iter.next() else {
180            return diff;
181        };
182
183        loop {
184            let common_len = cmp::min(lhs_span.text, rhs_span.text);
185
186            // Record a diff span if the color specs differ.
187            if lhs_span.style != rhs_span.style {
188                diff.differing_spans.push(DiffColorSpan {
189                    start: pos,
190                    len: common_len,
191                    lhs_color_spec: lhs_span.style,
192                    rhs_color_spec: rhs_span.style,
193                });
194            }
195
196            pos += common_len;
197
198            match lhs_span.text.cmp(&rhs_span.text) {
199                Ordering::Less => {
200                    rhs_span.text -= lhs_span.text;
201                    lhs_span = lhs_iter.next().unwrap();
202                    // ^ `unwrap()` here and below are safe; we've checked that
203                    // `lhs` and `rhs` contain same total span coverage.
204                }
205                Ordering::Greater => {
206                    lhs_span.text -= rhs_span.text;
207                    rhs_span = rhs_iter.next().unwrap();
208                }
209                Ordering::Equal => {
210                    lhs_span = match lhs_iter.next() {
211                        Some(span) => span,
212                        None => return diff,
213                    };
214                    rhs_span = match rhs_iter.next() {
215                        Some(span) => span,
216                        None => return diff,
217                    };
218                }
219            }
220        }
221    }
222
223    pub(crate) fn is_empty(&self) -> bool {
224        self.differing_spans.is_empty()
225    }

227    /// Highlights this diff on the specified `text` which has styling set with `color_spans`.
228    pub(crate) fn highlight_text(
229        &self,
230        out: &mut impl Write,
231        text: &str,
232        color_spans: &[StyledSpan<usize>],
233    ) -> io::Result<()> {
234        const SIDELINE_HL: Style = Style {
235            fg: Some(Color::RED),
236            ..Style::NONE
237        };
238
239        let highlights = HighlightedSpan::new(&self.differing_spans);
240        let mut highlights = highlights.iter().copied().peekable();
241        let mut line_start = 0;
242
243        // Spans together with their starting index
244        let mut span_start = 0;
245        let mut color_spans = color_spans
246            .iter()
247            .map(move |span| {
248                let prev_start = span_start;
249                span_start += span.text;
250                (prev_start, span)
251            })
252            .peekable();
253
254        for line in text.split('\n') {
255            let line_contains_spans = highlights
256                .peek()
257                .is_some_and(|span| span.start <= line_start + line.len());
258
259            if line_contains_spans {
260                write!(out, "{SIDELINE_HL}> {SIDELINE_HL:#}")?;
261                StyledSpan::write_line(&mut color_spans, out, line_start, line)?;
262                write!(out, "{SIDELINE_HL}> {SIDELINE_HL:#}")?;
263                Self::highlight_line(out, &mut highlights, line_start, line)?;
264            } else {
265                write!(out, "= ")?;
266                StyledSpan::write_line(&mut color_spans, out, line_start, line)?;
267            }
268            line_start += line.len() + 1;
269        }
270        Ok(())
271    }

273    fn highlight_line<I>(
274        out: &mut impl Write,
275        spans_iter: &mut Peekable<I>,
276        line_offset: usize,
277        line: &str,
278    ) -> io::Result<()>
279    where
280        I: Iterator<Item = HighlightedSpan>,
281    {
282        let line_len = line.len();
283        let mut line_pos = 0;
284
285        while line_pos < line_len {
286            let Some(span) = spans_iter.peek() else {
287                break;
288            };
289            let span_start = span.start.saturating_sub(line_offset);
290            if span_start >= line_len {
291                break;
292            }
293            let span_end = cmp::min(span.start + span.len - line_offset, line_len);
294
295            if span_start > line_pos {
296                let spaces = " ".repeat(line[line_pos..span_start].width());
297                write!(out, "{spaces}")?;
298            }
299
300            let ch = span.kind.underline_char();
301            let underline: String =
302                iter::repeat_n(ch, line[span_start..span_end].width()).collect();
303            let hl = span.kind.highlight_style();
304            write!(out, "{hl}{underline}{hl:#}")?;
305
306            line_pos = span_end;
307            if span.start + span.len <= line_offset + line_len {
308                // Span is finished on this line; can proceed to the next one.
309                spans_iter.next();
310            }
311        }
312        writeln!(out)
313    }

315    pub(crate) fn write_as_table(&self, out: &mut impl Write) -> io::Result<()> {
316        const TABLE_HEAD: Style = Style {
317            bold: true,
318            ..Style::NONE
319        };
320        const POS_WIDTH: usize = 10;
321
322        // Write table header.
323        writeln!(
324            out,
325            "{TABLE_HEAD}{pos:^POS_WIDTH$} {lhs:^STYLE_WIDTH$} {rhs:^STYLE_WIDTH$}",
326            pos = "Positions",
327            lhs = "Expected style",
328            rhs = "Actual style"
329        )?;
330        writeln!(
331            out,
332            "{pos:=>POS_WIDTH$} {lhs:=>STYLE_WIDTH$} {rhs:=>STYLE_WIDTH$}{TABLE_HEAD:#}",
333            pos = "",
334            lhs = "",
335            rhs = ""
336        )?;
337
338        // Write table itself.
339        for differing_span in &self.differing_spans {
340            let lhs_style = &differing_span.lhs_color_spec;
341            let mut lhs_lines = Self::write_style(lhs_style);
342            let rhs_style = &differing_span.rhs_color_spec;
343            let mut rhs_lines = Self::write_style(rhs_style);
344            if lhs_lines.len() < rhs_lines.len() {
345                lhs_lines.resize_with(rhs_lines.len(), String::new);
346            } else {
347                rhs_lines.resize_with(lhs_lines.len(), String::new);
348            }
349
350            for (i, (lhs_line, rhs_line)) in lhs_lines.into_iter().zip(rhs_lines).enumerate() {
351                if i == 0 {
352                    let start = differing_span.start;
353                    let end = start + differing_span.len;
354                    let pos = format!("{start}..{end}");
355                    write!(out, "{pos:>POS_WIDTH$} ")?;
356                } else {
357                    write!(out, "{:>POS_WIDTH$} ", "")?;
358                }
359                writeln!(
360                    out,
361                    "{lhs_style}{lhs_line:^STYLE_WIDTH$}{lhs_style:#} \
362                     {rhs_style}{rhs_line:^STYLE_WIDTH$}{rhs_style:#}"
363                )?;
364            }
365        }
366
367        Ok(())
368    }

370    /// Writes `color_spec` in human-readable format.
371    fn write_style(style: &Style) -> Vec<String> {
372        let mut tags = vec![];
373        if let Some(color) = style.fg {
374            tags.push(format!("fg:{}", Self::color_to_string(color)));
375        }
376        if let Some(color) = style.bg {
377            tags.push(format!("bg:{}", Self::color_to_string(color)));
378        }
379        if style.bold {
380            tags.push("bold".to_owned());
381        }
382        if style.italic {
383            tags.push("italic".to_owned());
384        }
385        if style.underline {
386            tags.push("underline".to_owned());
387        }
388        if style.strikethrough {
389            tags.push("strikethrough".to_owned());
390        }
391        if style.dimmed {
392            tags.push("dimmed".to_owned());
393        }
394        if style.blink {
395            tags.push("blink".to_owned());
396        }
397        if style.concealed {
398            tags.push("concealed".to_owned());
399        }
400
401        if tags.is_empty() {
402            tags.push("(none)".to_owned());
403        }
404
405        let mut lines = vec![];
406        let mut current_line = String::new();
407        for tag in tags {
408            // We can use `len()` because all text is ASCII.
409            if !current_line.is_empty() && current_line.len() + 1 + tag.len() > STYLE_WIDTH {
410                lines.push(mem::take(&mut current_line));
411            }
412            if !current_line.is_empty() {
413                current_line.push(' ');
414            }
415            current_line.push_str(&tag);
416        }
417
418        if !current_line.is_empty() {
419            lines.push(current_line);
420        }
421        lines
422    }

424    fn color_to_string(color: Color) -> String {
425        match color {
426            Color::Index(0) => "black".to_owned(),
427            Color::Index(1) => "red".to_owned(),
428            Color::Index(2) => "green".to_owned(),
429            Color::Index(3) => "yellow".to_owned(),
430            Color::Index(4) => "blue".to_owned(),
431            Color::Index(5) => "magenta".to_owned(),
432            Color::Index(6) => "cyan".to_owned(),
433            Color::Index(7) => "white".to_owned(),
434
435            Color::Index(8) => "black*".to_owned(),
436            Color::Index(9) => "red*".to_owned(),
437            Color::Index(10) => "green*".to_owned(),
438            Color::Index(11) => "yellow*".to_owned(),
439            Color::Index(12) => "blue*".to_owned(),
440            Color::Index(13) => "magenta*".to_owned(),
441            Color::Index(14) => "cyan*".to_owned(),
442            Color::Index(15) => "white*".to_owned(),
443
444            Color::Rgb(RgbColor(r, g, b)) => format!("#{r:02x}{g:02x}{b:02x}"),
445
446            Color::Index(_) => unreachable!(), // must be transformed during color normalization
447        }
448    }

451#[derive(Debug, Clone, Copy)]
452enum SpanHighlightKind {
453    Main,
454    Aux,
455}
456
457impl SpanHighlightKind {
458    fn underline_char(self) -> char {
459        match self {
460            Self::Main => '^',
461            Self::Aux => '!',
462        }
463    }
464
465    fn highlight_style(self) -> Style {
466        let mut style = Style::default();
467        match self {
468            Self::Main => {
469                style.fg = Some(Color::WHITE);
470                style.bg = Some(Color::RED);
471            }
472            Self::Aux => {
473                style.fg = Some(Color::BLACK);
474                style.bg = Some(Color::YELLOW);
475            }
476        }
477        style
478    }

481#[derive(Debug, Clone, Copy)]
482struct HighlightedSpan {
483    start: usize,
484    len: usize,
485    kind: SpanHighlightKind,
486}
487
488impl HighlightedSpan {
489    fn new(differing_spans: &[DiffColorSpan]) -> Vec<Self> {
490        let mut sequential_span_count = 1;
491        let span_highlights = differing_spans.windows(2).map(|window| match window {
492            [prev, next] => {
493                if prev.start + prev.len == next.start {
494                    sequential_span_count += 1;
495                } else {
496                    sequential_span_count = 1;
497                }
498                if sequential_span_count % 2 == 0 {
499                    SpanHighlightKind::Aux
500                } else {
501                    SpanHighlightKind::Main
502                }
503            }
504            _ => unreachable!(),
505        });
506
507        iter::once(SpanHighlightKind::Main)
508            .chain(span_highlights)
509            .zip(differing_spans)
510            .map(|(kind, span)| Self {
511                start: span.start,
512                len: span.len,
513                kind,
514            })
515            .collect()
516    }
