1//! Parser for terminal output that converts it to a sequence of instructions to
2//! a writer implementing `WriteColor`.
3
4use std::str;
5
6use crate::{
7    style::{Color, RgbColor, Style, WriteStyled},
8    TermError,
9};
10
11const ANSI_ESC: u8 = 0x1b;
12const ANSI_BEL: u8 = 0x07;
13const ANSI_CSI: u8 = b'[';
14const ANSI_OCS: u8 = b']';
15
16/// Parses terminal output and issues corresponding commands to the `writer`.
17#[derive(Debug)]
18pub(crate) struct TermOutputParser<'a, W: ?Sized> {
19    writer: &'a mut W,
20    style: Style,
21}
22
23impl<'a, W: WriteStyled + ?Sized> TermOutputParser<'a, W> {
24    pub(crate) fn new(writer: &'a mut W) -> Self {
25        Self {
26            writer,
27            style: Style::default(),
28        }
29    }
30
31    /// Handles an operating system command. Skip all chars until BEL (\u{7}) or ST (\u{1b}).
32    fn skip_ocs(term_output: &[u8], i: &mut usize) -> Result<(), TermError> {
33        while *i < term_output.len() && term_output[*i] != ANSI_BEL && term_output[*i] != ANSI_ESC {
34            *i += 1;
35        }
36
37        if *i == term_output.len() {
38            return Err(TermError::UnfinishedSequence);
39        }
40        if term_output[*i] == ANSI_ESC {
41            *i += 1;
42            if *i == term_output.len() {
43                return Err(TermError::UnfinishedSequence);
44            }
45            if term_output[*i] != b'\\' {
46                return Err(TermError::UnrecognizedSequence(term_output[*i]));
47            }
48        }
49        *i += 1;
50        Ok(())
51    }
52
53    /// Checks whether the given chunk of text has any non-escaped parts.
54    fn has_plaintext(bytes: &[u8]) -> Result<bool, TermError> {
55        let mut i = 0;
56        while i < bytes.len() {
57            if bytes[i] != ANSI_ESC {
58                return Ok(true);
59            }
60
61            i += 1;
62            let next_byte = bytes.get(i).copied().ok_or(TermError::UnfinishedSequence)?;
63            if next_byte == ANSI_CSI {
64                i += 1;
65                let csi = Csi::parse(&bytes[i..])?;
66                i += csi.len;
67            } else if next_byte == ANSI_OCS {
68                Self::skip_ocs(bytes, &mut i)?;
69            } else {
70                return Err(TermError::UnrecognizedSequence(next_byte));
71            }
72        }
73        Ok(false)
74    }
75
76    pub(crate) fn parse(&mut self, term_output: &[u8]) -> Result<(), TermError> {
77        let lines: Vec<_> = term_output.split(|&ch| ch == b'\n').collect();
78        let line_count = lines.len();
79
80        for (i, line) in lines.into_iter().enumerate() {
81            // Find the last occurrence of `\r` that has text output after it, and trim everything before it.
82            // This works reasonably well in most cases.
83            let chunks = line.rsplit(|&ch| ch == b'\r');
84            let mut processed_len = 0;
85            for chunk in chunks {
86                processed_len += chunk.len() + 1;
87                if Self::has_plaintext(chunk)? {
88                    break;
89                }
90            }
91            let start_pos = line.len().saturating_sub(processed_len);
92            let processed_line = &line[start_pos..];
93
94            self.parse_line(processed_line)?;
95
96            if i + 1 < line_count {
97                self.writer.write_text("\n").map_err(TermError::Io)?;
98            }
99        }
100        Ok(())
101    }
102
103    fn parse_line(&mut self, term_output: &[u8]) -> Result<(), TermError> {
104        let mut dirty_color_spec = false;
105
106        let mut i = 0;
107        let mut written_end = 0;
108        while i < term_output.len() {
109            if term_output[i] == ANSI_ESC {
110                // Push the preceding "ordinary" bytes into the writer.
111                self.write_ordinary_text(&term_output[written_end..i], &mut dirty_color_spec)?;
112
113                i += 1;
114                let next_byte = term_output
115                    .get(i)
116                    .copied()
117                    .ok_or(TermError::UnfinishedSequence)?;
118                if next_byte == ANSI_CSI {
119                    i += 1;
120                    let csi = Csi::parse(&term_output[i..])?;
121                    let prev_color_spec = self.style;
122                    csi.update_color_spec(&mut self.style)?;
123                    dirty_color_spec = dirty_color_spec || prev_color_spec != self.style;
124                    i += csi.len;
125                } else if next_byte == ANSI_OCS {
126                    Self::skip_ocs(term_output, &mut i)?;
127                } else {
128                    return Err(TermError::UnrecognizedSequence(next_byte));
129                }
130                written_end = i; // skip the escape sequence
131            } else if term_output[i] == b'\r' {
132                self.write_ordinary_text(&term_output[written_end..i], &mut dirty_color_spec)?;
133                i += 1;
134                written_end = i; // skip writing '\r'
135            } else {
136                // Ordinary char.
137                i += 1;
138            }
139        }
140
141        // We write the terminal color spec even if the text is empty.
142        if dirty_color_spec {
143            self.writer
144                .write_style(&self.style)
145                .map_err(TermError::Io)?;
146        }
147
148        let remaining_text = str::from_utf8(&term_output[written_end..i])?;
149        self.writer
150            .write_text(remaining_text)
151            .map_err(TermError::Io)
152    }
153
154    fn write_ordinary_text(
155        &mut self,
156        text: &[u8],
157        dirty_color_spec: &mut bool,
158    ) -> Result<(), TermError> {
159        if text.is_empty() {
160            Ok(())
161        } else {
162            let text = str::from_utf8(text)?;
163            if *dirty_color_spec {
164                *dirty_color_spec = false;
165                self.writer
166                    .write_style(&self.style)
167                    .map_err(TermError::Io)?;
168            }
169            self.writer.write_text(text).map_err(TermError::Io)
170        }
171    }
172}

174#[derive(Debug, Clone, Copy)]
175struct Csi<'a> {
176    parameters: &'a [u8],
177    final_byte: u8,
178    len: usize,
179}
180
181impl<'a> Csi<'a> {
182    fn parse(buffer: &'a [u8]) -> Result<Self, TermError> {
183        let intermediates_start = buffer
184            .iter()
185            .position(|byte| !(0x30..=0x3f).contains(byte))
186            .ok_or(TermError::UnfinishedSequence)?;
187
188        let final_byte_pos = buffer[intermediates_start..]
189            .iter()
190            .position(|byte| !(0x20..=0x2f).contains(byte))
191            .ok_or(TermError::UnfinishedSequence)?;
192        let final_byte_pos = intermediates_start + final_byte_pos;
193
194        let final_byte = buffer[final_byte_pos];
195        if (0x40..=0x7e).contains(&final_byte) {
196            Ok(Self {
197                parameters: &buffer[..intermediates_start],
198                final_byte,
199                len: final_byte_pos + 1,
200            })
201        } else {
202            Err(TermError::InvalidSgrFinalByte(final_byte))
203        }
204    }
205
206    fn update_color_spec(self, spec: &mut Style) -> Result<(), TermError> {
207        if self.final_byte != b'm' {
208            return Ok(());
209        }
210
211        let mut params = self.parameters.split(|&byte| byte == b';').peekable();
212        if params.peek().is_none() {
213            *spec = Style::default(); // reset
214        }
215        while params.peek().is_some() {
216            Self::process_param(spec, &mut params)?;
217        }
218        Ok(())
219    }
220
221    fn process_param(
222        style: &mut Style,
223        mut params: impl Iterator<Item = &'a [u8]>,
224    ) -> Result<(), TermError> {
225        let param = params.next().unwrap();
226        if let Some(fg_color) = Self::parse_simple_fg_color(param) {
227            style.fg = Some(fg_color);
228        } else if let Some(bg_color) = Self::parse_simple_bg_color(param) {
229            style.bg = Some(bg_color);
230        } else {
231            match param {
232                b"" | b"0" => {
233                    *style = Style::default();
234                }
235                b"1" => {
236                    style.bold = true;
237                }
238                b"2" => {
239                    style.dimmed = true;
240                }
241                b"3" => {
242                    style.italic = true;
243                }
244                b"4" => {
245                    style.underline = true;
246                }
247                b"5" | b"6" => {
248                    style.blink = true;
249                }
250                b"7" => {
251                    style.inverted = true;
252                }
253                b"8" => {
254                    style.concealed = true;
255                }
256                b"9" => {
257                    style.strikethrough = true;
258                }
259
260                b"22" => {
261                    style.bold = false;
262                    style.dimmed = false;
263                }
264                b"23" => {
265                    style.italic = false;
266                }
267                b"24" => {
268                    style.underline = false;
269                }
270                b"25" => {
271                    style.blink = false;
272                }
273                b"27" => {
274                    style.inverted = false;
275                }
276                b"28" => {
277                    style.concealed = false;
278                }
279                b"29" => {
280                    style.strikethrough = false;
281                }
282
283                // Compound foreground color spec
284                b"38" => {
285                    let color = Self::read_color(params)?;
286                    style.fg = Some(color);
287                }
288                b"39" => {
289                    style.fg = None;
290                }
291                // Compound background color spec
292                b"48" => {
293                    let color = Self::read_color(params)?;
294                    style.bg = Some(color);
295                }
296                b"49" => {
297                    style.bg = None;
298                }
299
300                _ => { /* Do nothing */ }
301            }
302        }
303        Ok(())
304    }
305
306    fn parse_simple_fg_color(param: &[u8]) -> Option<Color> {
307        Some(match param {
308            b"30" => Color::BLACK,
309            b"31" => Color::RED,
310            b"32" => Color::GREEN,
311            b"33" => Color::YELLOW,
312            b"34" => Color::BLUE,
313            b"35" => Color::MAGENTA,
314            b"36" => Color::CYAN,
315            b"37" => Color::WHITE,
316
317            b"90" => Color::INTENSE_BLACK,
318            b"91" => Color::INTENSE_RED,
319            b"92" => Color::INTENSE_GREEN,
320            b"93" => Color::INTENSE_YELLOW,
321            b"94" => Color::INTENSE_BLUE,
322            b"95" => Color::INTENSE_MAGENTA,
323            b"96" => Color::INTENSE_CYAN,
324            b"97" => Color::INTENSE_WHITE,
325
326            _ => return None,
327        })
328    }
329
330    fn parse_simple_bg_color(param: &[u8]) -> Option<Color> {
331        Some(match param {
332            b"40" => Color::BLACK,
333            b"41" => Color::RED,
334            b"42" => Color::GREEN,
335            b"43" => Color::YELLOW,
336            b"44" => Color::BLUE,
337            b"45" => Color::MAGENTA,
338            b"46" => Color::CYAN,
339            b"47" => Color::WHITE,
340
341            b"100" => Color::INTENSE_BLACK,
342            b"101" => Color::INTENSE_RED,
343            b"102" => Color::INTENSE_GREEN,
344            b"103" => Color::INTENSE_YELLOW,
345            b"104" => Color::INTENSE_BLUE,
346            b"105" => Color::INTENSE_MAGENTA,
347            b"106" => Color::INTENSE_CYAN,
348            b"107" => Color::INTENSE_WHITE,
349
350            _ => return None,
351        })
352    }
353
354    fn read_color(mut params: impl Iterator<Item = &'a [u8]>) -> Result<Color, TermError> {
355        let color_type = params.next().ok_or(TermError::UnfinishedColor)?;
356        match color_type {
357            b"5" => {
358                let index = params.next().ok_or(TermError::UnfinishedColor)?;
359                Self::parse_color_index(index).map(Color::Index)
360            }
361            b"2" => {
362                let r = params.next().ok_or(TermError::UnfinishedColor)?;
363                let g = params.next().ok_or(TermError::UnfinishedColor)?;
364                let b = params.next().ok_or(TermError::UnfinishedColor)?;
365
366                let r = Self::parse_color_index(r)?;
367                let g = Self::parse_color_index(g)?;
368                let b = Self::parse_color_index(b)?;
369                Ok(Color::Rgb(RgbColor(r, g, b)))
370            }
371            _ => {
372                let color_type = String::from_utf8_lossy(color_type).into_owned();
373                Err(TermError::InvalidColorType(color_type))
374            }
375        }
376    }
377
378    fn parse_color_index(param: &[u8]) -> Result<u8, TermError> {
379        if param.is_empty() {
380            // As per ANSI standards, empty params are treated as number 0.
381            return Ok(0);
382        }
383
384        let param = unsafe {
385            // SAFETY: safe by construction; we've checked range of bytes in params
386            // when creating a `Csi` instance.
387            str::from_utf8_unchecked(param)
388        };
389        param.parse().map_err(TermError::InvalidColorIndex)
390    }
