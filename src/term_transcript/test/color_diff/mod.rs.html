16pub(crate) struct ColorSpan {
17    pub(crate) len: usize,
18    color_spec: ColorSpec,
19}

21impl ColorSpan {
22    pub fn parse(ansi_text: &str) -> Result<Vec<Self>, TermError> {
23        let mut spans = ColorSpansWriter::default();
24        TermOutputParser::new(&mut spans).parse(ansi_text.as_bytes())?;
25        Ok(spans.shrink().spans)
26    }
27
28    pub fn write_colorized(
29        spans: &[Self],
30        out: &mut impl WriteColor,
31        plaintext: &str,
32    ) -> io::Result<()> {
33        debug_assert_eq!(
34            spans.iter().map(|span| span.len).sum::<usize>(),
35            plaintext.len()
36        );
37        let mut pos = 0;
38        for span in spans {
39            out.set_color(&span.color_spec)?;
40            write!(out, "{}", &plaintext[pos..pos + span.len])?;
41            pos += span.len;
42        }
43        Ok(())
44    }
45
46    /// Writes a single plaintext `line` to `out` using styles from `spans_iter`.
47    fn write_line<'a, I: Iterator<Item = (usize, &'a Self)>>(
48        spans_iter: &mut Peekable<I>,
49        out: &mut impl WriteColor,
50        line_start: usize,
51        line: &str,
52    ) -> io::Result<()> {
53        let mut pos = 0;
54        while pos < line.len() {
55            let &(span_start, span) = spans_iter.peek().expect("spans ended before lines");
56            let span_len = span.len - line_start.saturating_sub(span_start);
57
58            let span_end = cmp::min(pos + span_len, line.len());
59            out.set_color(&span.color_spec)?;
60            write!(out, "{}", &line[pos..span_end])?;
61            if span_end == pos + span_len {
62                // The span has ended, can proceed to the next one.
63                spans_iter.next();
64            }
65            pos += span_len;
66        }
67        out.reset()?;
68        writeln!(out)
69    }
70}

74pub(crate) struct ColorSpansWriter {
75    spans: Vec<ColorSpan>,
76    color_spec: ColorSpec,
77}

79impl ColorSpansWriter {
80    fn normalize_spec(mut spec: ColorSpec) -> ColorSpec {
81        if let Some(color) = spec.fg().copied() {
82            spec.set_fg(Some(Self::normalize_color(color)));
83        }
84        if let Some(color) = spec.bg().copied() {
85            spec.set_bg(Some(Self::normalize_color(color)));
86        }
87        spec
88    }
89
90    fn normalize_color(color: Color) -> Color {
91        match color {
92            Color::Ansi256(0) => Color::Black,
93            Color::Ansi256(1) => Color::Red,
94            Color::Ansi256(2) => Color::Green,
95            Color::Ansi256(3) => Color::Yellow,
96            Color::Ansi256(4) => Color::Blue,
97            Color::Ansi256(5) => Color::Magenta,
98            Color::Ansi256(6) => Color::Cyan,
99            Color::Ansi256(7) => Color::White,
100
101            Color::Ansi256(index) if index >= 16 => match IndexOrRgb::indexed_color(index) {
102                IndexOrRgb::Rgb(r, g, b) => Color::Rgb(r, g, b),
103                IndexOrRgb::Index(_) => color,
104            },
105
106            _ => color,
107        }
108    }
109
110    /// Unites sequential spans with the same color spec.
111    fn shrink(self) -> Self {
112        let mut shrunk_spans = Vec::<ColorSpan>::with_capacity(self.spans.len());
113        for span in self.spans {
114            if let Some(last_span) = shrunk_spans.last_mut() {
115                if last_span.color_spec == span.color_spec {
116                    last_span.len += span.len;
117                } else {
118                    shrunk_spans.push(span);
119                }
120            } else {
121                shrunk_spans.push(span);
122            }
123        }
124
125        Self {
126            spans: shrunk_spans,
127            color_spec: self.color_spec,
128        }
129    }
130
131    pub fn into_inner(self) -> Vec<ColorSpan> {
132        self.shrink().spans
133    }
134}

175#[derive(Debug, Default)]
176pub(crate) struct ColorDiff {
177    differing_spans: Vec<DiffColorSpan>,
178}
179
180impl ColorDiff {
181    pub fn new(lhs: &[ColorSpan], rhs: &[ColorSpan]) -> Self {
182        debug_assert_eq!(
183            lhs.iter().map(|span| span.len).sum::<usize>(),
184            rhs.iter().map(|span| span.len).sum::<usize>(),
185            "Spans {lhs:?} and {rhs:?} must have equal total covered length"
186        );
187
188        let mut diff = Self::default();
189        let mut pos = 0;
190        let mut lhs_iter = lhs.iter().cloned();
191        let Some(mut lhs_span) = lhs_iter.next() else {
192            return diff;
193        };
194        let mut rhs_iter = rhs.iter().cloned();
195        let Some(mut rhs_span) = rhs_iter.next() else {
196            return diff;
197        };
198
199        loop {
200            let common_len = cmp::min(lhs_span.len, rhs_span.len);
201
202            // Record a diff span if the color specs differ.
203            if lhs_span.color_spec != rhs_span.color_spec {
204                diff.differing_spans.push(DiffColorSpan {
205                    start: pos,
206                    len: common_len,
207                    lhs_color_spec: lhs_span.color_spec.clone(),
208                    rhs_color_spec: rhs_span.color_spec.clone(),
209                });
210            }
211
212            pos += common_len;
213
214            match lhs_span.len.cmp(&rhs_span.len) {
215                Ordering::Less => {
216                    rhs_span.len -= lhs_span.len;
217                    lhs_span = lhs_iter.next().unwrap();
218                    // ^ `unwrap()` here and below are safe; we've checked that
219                    // `lhs` and `rhs` contain same total span coverage.
220                }
221                Ordering::Greater => {
222                    lhs_span.len -= rhs_span.len;
223                    rhs_span = rhs_iter.next().unwrap();
224                }
225                Ordering::Equal => {
226                    lhs_span = match lhs_iter.next() {
227                        Some(span) => span,
228                        None => return diff,
229                    };
230                    rhs_span = match rhs_iter.next() {
231                        Some(span) => span,
232                        None => return diff,
233                    };
234                }
235            }
236        }
237    }
238
239    pub fn is_empty(&self) -> bool {
240        self.differing_spans.is_empty()
241    }

243    /// Highlights this diff on the specified `text` which has styling set with `color_spans`.
244    pub fn highlight_text(
245        &self,
246        out: &mut impl WriteColor,
247        text: &str,
248        color_spans: &[ColorSpan],
249    ) -> io::Result<()> {
250        let mut sideline_hl = ColorSpec::new();
251        sideline_hl.set_fg(Some(Color::Red));
252
253        let highlights = HighlightedSpan::new(&self.differing_spans);
254        let mut highlights = highlights.iter().copied().peekable();
255        let mut line_start = 0;
256
257        // Spans together with their starting index
258        let mut span_start = 0;
259        let mut color_spans = color_spans
260            .iter()
261            .map(move |span| {
262                let prev_start = span_start;
263                span_start += span.len;
264                (prev_start, span)
265            })
266            .peekable();
267
268        for line in text.split('\n') {
269            let line_contains_spans = highlights
270                .peek()
271                .is_some_and(|span| span.start <= line_start + line.len());
272
273            if line_contains_spans {
274                out.set_color(&sideline_hl)?;
275                write!(out, "> ")?;
276                out.reset()?;
277                ColorSpan::write_line(&mut color_spans, out, line_start, line)?;
278                out.set_color(&sideline_hl)?;
279                write!(out, "> ")?;
280                out.reset()?;
281                Self::highlight_line(out, &mut highlights, line_start, line)?;
282            } else {
283                write!(out, "= ")?;
284                ColorSpan::write_line(&mut color_spans, out, line_start, line)?;
285            }
286            line_start += line.len() + 1;
287        }
288        Ok(())
289    }

291    fn highlight_line<I: Iterator<Item = HighlightedSpan>>(
292        out: &mut impl WriteColor,
293        spans_iter: &mut Peekable<I>,
294        line_offset: usize,
295        line: &str,
296    ) -> io::Result<()> {
297        let line_len = line.len();
298        let mut line_pos = 0;
299
300        while line_pos < line_len {
301            let Some(span) = spans_iter.peek() else {
302                break;
303            };
304            let span_start = span.start.saturating_sub(line_offset);
305            if span_start >= line_len {
306                break;
307            }
308            let span_end = cmp::min(span.start + span.len - line_offset, line_len);
309
310            if span_start > line_pos {
311                let spaces = " ".repeat(line[line_pos..span_start].width());
312                write!(out, "{spaces}")?;
313            }
314
315            let ch = span.kind.underline_char();
316            let underline: String =
317                iter::repeat_n(ch, line[span_start..span_end].width()).collect();
318            out.set_color(&span.kind.highlight_spec())?;
319            write!(out, "{underline}")?;
320            out.reset()?;
321
322            line_pos = span_end;
323            if span.start + span.len <= line_offset + line_len {
324                // Span is finished on this line; can proceed to the next one.
325                spans_iter.next();
326            }
327        }
328        writeln!(out)
329    }

331    pub fn write_as_table(&self, out: &mut impl WriteColor) -> io::Result<()> {
332        const POS_WIDTH: usize = 10;
333        const STYLE_WIDTH: usize = 22; // `buid magenta*/magenta*`
334
335        // Write table header.
336        let mut table_header_spec = ColorSpec::new();
337        table_header_spec.set_bold(true);
338        table_header_spec.set_intense(true);
339        out.set_color(&table_header_spec)?;
340        writeln!(
341            out,
342            "{pos:^POS_WIDTH$} {lhs:^STYLE_WIDTH$} {rhs:^STYLE_WIDTH$}",
343            pos = "Positions",
344            lhs = "Expected style",
345            rhs = "Actual style"
346        )?;
347        writeln!(
348            out,
349            "{pos:=>POS_WIDTH$} {lhs:=>STYLE_WIDTH$} {rhs:=>STYLE_WIDTH$}",
350            pos = "",
351            lhs = "",
352            rhs = ""
353        )?;
354        out.reset()?;
355
356        // Write table itself.
357        for differing_span in &self.differing_spans {
358            let start = differing_span.start;
359            let end = start + differing_span.len;
360            let pos = format!("{start}..{end}");
361            write!(out, "{pos:>POS_WIDTH$} ")?;
362
363            Self::write_color_spec(out, &differing_span.lhs_color_spec)?;
364            out.write_all(b" ")?;
365            Self::write_color_spec(out, &differing_span.rhs_color_spec)?;
366            writeln!(out)?;
367        }
368
369        Ok(())
370    }

372    /// Writes `color_spec` in human-readable format.
373    fn write_color_spec(out: &mut impl WriteColor, color_spec: &ColorSpec) -> io::Result<()> {
374        const COLOR_WIDTH: usize = 8; // `magenta*` is the widest color output
375
376        out.set_color(color_spec)?;
377        out.write_all(if color_spec.bold() { b"b" } else { b"-" })?;
378        out.write_all(if color_spec.italic() { b"i" } else { b"-" })?;
379        out.write_all(if color_spec.underline() { b"u" } else { b"-" })?;
380        out.write_all(if color_spec.dimmed() { b"d" } else { b"-" })?;
381
382        write!(
383            out,
384            " {fg:>COLOR_WIDTH$}/{bg:<COLOR_WIDTH$}",
385            fg = color_spec
386                .fg()
387                .map_or_else(|| "(none)".to_owned(), |&fg| Self::color_to_string(fg)),
388            bg = color_spec
389                .bg()
390                .map_or_else(|| "(none)".to_owned(), |&bg| Self::color_to_string(bg)),
391        )?;
392        out.reset()
393    }

395    fn color_to_string(color: Color) -> String {
396        match color {
397            Color::Black | Color::Ansi256(0) => "black".to_owned(),
398            Color::Red | Color::Ansi256(1) => "red".to_owned(),
399            Color::Green | Color::Ansi256(2) => "green".to_owned(),
400            Color::Yellow | Color::Ansi256(3) => "yellow".to_owned(),
401            Color::Blue | Color::Ansi256(4) => "blue".to_owned(),
402            Color::Magenta | Color::Ansi256(5) => "magenta".to_owned(),
403            Color::Cyan | Color::Ansi256(6) => "cyan".to_owned(),
404            Color::White | Color::Ansi256(7) => "white".to_owned(),
405
406            Color::Ansi256(8) => "black*".to_owned(),
407            Color::Ansi256(9) => "red*".to_owned(),
408            Color::Ansi256(10) => "green*".to_owned(),
409            Color::Ansi256(11) => "yellow*".to_owned(),
410            Color::Ansi256(12) => "blue*".to_owned(),
411            Color::Ansi256(13) => "magenta*".to_owned(),
412            Color::Ansi256(14) => "cyan*".to_owned(),
413            Color::Ansi256(15) => "white*".to_owned(),
414
415            Color::Rgb(r, g, b) => format!("#{r:02x}{g:02x}{b:02x}"),
416
417            _ => unreachable!(), // must be transformed during color normalization
418        }
419    }

422#[derive(Debug, Clone, Copy)]
423enum SpanHighlightKind {
424    Main,
425    Aux,
426}
427
428impl SpanHighlightKind {
429    fn underline_char(self) -> char {
430        match self {
431            Self::Main => '^',
432            Self::Aux => '!',
433        }
434    }
435
436    fn highlight_spec(self) -> ColorSpec {
437        let mut spec = ColorSpec::new();
438        match self {
439            Self::Main => {
440                spec.set_fg(Some(Color::White)).set_bg(Some(Color::Red));
441            }
442            Self::Aux => {
443                spec.set_fg(Some(Color::Black)).set_bg(Some(Color::Yellow));
444            }
445        }
446        spec
447    }

450#[derive(Debug, Clone, Copy)]
451struct HighlightedSpan {
452    start: usize,
453    len: usize,
454    kind: SpanHighlightKind,
455}
456
457impl HighlightedSpan {
458    fn new(differing_spans: &[DiffColorSpan]) -> Vec<Self> {
459        let mut sequential_span_count = 1;
460        let span_highlights = differing_spans.windows(2).map(|window| match window {
461            [prev, next] => {
462                if prev.start + prev.len == next.start {
463                    sequential_span_count += 1;
464                } else {
465                    sequential_span_count = 1;
466                }
467                if sequential_span_count % 2 == 0 {
468                    SpanHighlightKind::Aux
469                } else {
470                    SpanHighlightKind::Main
471                }
472            }
473            _ => unreachable!(),
474        });
475
476        iter::once(SpanHighlightKind::Main)
477            .chain(span_highlights)
478            .zip(differing_spans)
479            .map(|(kind, span)| Self {
480                start: span.start,
481                len: span.len,
482                kind,
483            })
484            .collect()
485    }
