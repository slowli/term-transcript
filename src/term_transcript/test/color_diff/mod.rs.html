20pub(crate) struct ColorSpan {
21    pub(crate) len: usize,
22    color_spec: ColorSpec,
23}
24
25impl ColorSpan {
26    pub fn parse(ansi_text: &str) -> Result<Vec<Self>, TermError> {
27        let mut spans = ColorSpansWriter::default();
28        TermOutputParser::new(&mut spans).parse(ansi_text.as_bytes())?;
29        Ok(spans.shrink().spans)
30    }
31
32    pub fn write_colorized(
33        spans: &[Self],
34        out: &mut impl WriteColor,
35        plaintext: &str,
36    ) -> io::Result<()> {
37        debug_assert_eq!(
38            spans.iter().map(|span| span.len).sum::<usize>(),
39            plaintext.len()
40        );
41        let mut pos = 0;
42        for span in spans {
43            out.set_color(&span.color_spec)?;
44            write!(out, "{}", &plaintext[pos..pos + span.len])?;
45            pos += span.len;
46        }
47        Ok(())
48    }
49
50    /// Writes a single plaintext `line` to `out` using styles from `spans_iter`.
51    fn write_line<'a, I: Iterator<Item = (usize, &'a Self)>>(
52        spans_iter: &mut Peekable<I>,
53        out: &mut impl WriteColor,
54        line_start: usize,
55        line: &str,
56    ) -> io::Result<()> {
57        let mut pos = 0;
58        while pos < line.len() {
59            let &(span_start, span) = spans_iter.peek().expect("spans ended before lines");
60            let span_len = span.len - line_start.saturating_sub(span_start);
61
62            let span_end = cmp::min(pos + span_len, line.len());
63            out.set_color(&span.color_spec)?;
64            write!(out, "{}", &line[pos..span_end])?;
65            if span_end == pos + span_len {
66                // The span has ended, can proceed to the next one.
67                spans_iter.next();
68            }
69            pos += span_len;
70        }
71        out.reset()?;
72        writeln!(out)
73    }
74}

76/// `Write` / `WriteColor` implementation recording `ColorSpan`s for the input text.
77#[derive(Debug, Default)]
78pub(crate) struct ColorSpansWriter {
79    spans: Vec<ColorSpan>,
80    color_spec: ColorSpec,
81}

83impl ColorSpansWriter {
84    fn normalize_spec(mut spec: ColorSpec) -> ColorSpec {
85        if let Some(color) = spec.fg().copied() {
86            spec.set_fg(Some(Self::normalize_color(color)));
87        }
88        if let Some(color) = spec.bg().copied() {
89            spec.set_bg(Some(Self::normalize_color(color)));
90        }
91        spec
92    }
93
94    fn normalize_color(color: Color) -> Color {
95        match color {
96            Color::Ansi256(0) => Color::Black,
97            Color::Ansi256(1) => Color::Red,
98            Color::Ansi256(2) => Color::Green,
99            Color::Ansi256(3) => Color::Yellow,
100            Color::Ansi256(4) => Color::Blue,
101            Color::Ansi256(5) => Color::Magenta,
102            Color::Ansi256(6) => Color::Cyan,
103            Color::Ansi256(7) => Color::White,
104
105            Color::Ansi256(index) if index >= 16 => match IndexOrRgb::indexed_color(index) {
106                IndexOrRgb::Rgb(RgbColor(r, g, b)) => Color::Rgb(r, g, b),
107                IndexOrRgb::Index(_) => color,
108            },
109
110            _ => color,
111        }
112    }
113
114    /// Unites sequential spans with the same color spec.
115    fn shrink(self) -> Self {
116        let mut shrunk_spans = Vec::<ColorSpan>::with_capacity(self.spans.len());
117        for span in self.spans {
118            if let Some(last_span) = shrunk_spans.last_mut() {
119                if last_span.color_spec == span.color_spec {
120                    last_span.len += span.len;
121                } else {
122                    shrunk_spans.push(span);
123                }
124            } else {
125                shrunk_spans.push(span);
126            }
127        }
128
129        Self {
130            spans: shrunk_spans,
131            color_spec: self.color_spec,
132        }
133    }
134
135    pub fn into_inner(self) -> Vec<ColorSpan> {
136        self.shrink().spans
137    }
138}

179#[derive(Debug, Default)]
180pub(crate) struct ColorDiff {
181    differing_spans: Vec<DiffColorSpan>,
182}
183
184impl ColorDiff {
185    pub fn new(lhs: &[ColorSpan], rhs: &[ColorSpan]) -> Self {
186        debug_assert_eq!(
187            lhs.iter().map(|span| span.len).sum::<usize>(),
188            rhs.iter().map(|span| span.len).sum::<usize>(),
189            "Spans {lhs:?} and {rhs:?} must have equal total covered length"
190        );
191
192        let mut diff = Self::default();
193        let mut pos = 0;
194        let mut lhs_iter = lhs.iter().cloned();
195        let Some(mut lhs_span) = lhs_iter.next() else {
196            return diff;
197        };
198        let mut rhs_iter = rhs.iter().cloned();
199        let Some(mut rhs_span) = rhs_iter.next() else {
200            return diff;
201        };
202
203        loop {
204            let common_len = cmp::min(lhs_span.len, rhs_span.len);
205
206            // Record a diff span if the color specs differ.
207            if lhs_span.color_spec != rhs_span.color_spec {
208                diff.differing_spans.push(DiffColorSpan {
209                    start: pos,
210                    len: common_len,
211                    lhs_color_spec: lhs_span.color_spec.clone(),
212                    rhs_color_spec: rhs_span.color_spec.clone(),
213                });
214            }
215
216            pos += common_len;
217
218            match lhs_span.len.cmp(&rhs_span.len) {
219                Ordering::Less => {
220                    rhs_span.len -= lhs_span.len;
221                    lhs_span = lhs_iter.next().unwrap();
222                    // ^ `unwrap()` here and below are safe; we've checked that
223                    // `lhs` and `rhs` contain same total span coverage.
224                }
225                Ordering::Greater => {
226                    lhs_span.len -= rhs_span.len;
227                    rhs_span = rhs_iter.next().unwrap();
228                }
229                Ordering::Equal => {
230                    lhs_span = match lhs_iter.next() {
231                        Some(span) => span,
232                        None => return diff,
233                    };
234                    rhs_span = match rhs_iter.next() {
235                        Some(span) => span,
236                        None => return diff,
237                    };
238                }
239            }
240        }
241    }
242
243    pub fn is_empty(&self) -> bool {
244        self.differing_spans.is_empty()
245    }

247    /// Highlights this diff on the specified `text` which has styling set with `color_spans`.
248    pub fn highlight_text(
249        &self,
250        out: &mut impl WriteColor,
251        text: &str,
252        color_spans: &[ColorSpan],
253    ) -> io::Result<()> {
254        let mut sideline_hl = ColorSpec::new();
255        sideline_hl.set_fg(Some(Color::Red));
256
257        let highlights = HighlightedSpan::new(&self.differing_spans);
258        let mut highlights = highlights.iter().copied().peekable();
259        let mut line_start = 0;
260
261        // Spans together with their starting index
262        let mut span_start = 0;
263        let mut color_spans = color_spans
264            .iter()
265            .map(move |span| {
266                let prev_start = span_start;
267                span_start += span.len;
268                (prev_start, span)
269            })
270            .peekable();
271
272        for line in text.split('\n') {
273            let line_contains_spans = highlights
274                .peek()
275                .is_some_and(|span| span.start <= line_start + line.len());
276
277            if line_contains_spans {
278                out.set_color(&sideline_hl)?;
279                write!(out, "> ")?;
280                out.reset()?;
281                ColorSpan::write_line(&mut color_spans, out, line_start, line)?;
282                out.set_color(&sideline_hl)?;
283                write!(out, "> ")?;
284                out.reset()?;
285                Self::highlight_line(out, &mut highlights, line_start, line)?;
286            } else {
287                write!(out, "= ")?;
288                ColorSpan::write_line(&mut color_spans, out, line_start, line)?;
289            }
290            line_start += line.len() + 1;
291        }
292        Ok(())
293    }

295    fn highlight_line<I: Iterator<Item = HighlightedSpan>>(
296        out: &mut impl WriteColor,
297        spans_iter: &mut Peekable<I>,
298        line_offset: usize,
299        line: &str,
300    ) -> io::Result<()> {
301        let line_len = line.len();
302        let mut line_pos = 0;
303
304        while line_pos < line_len {
305            let Some(span) = spans_iter.peek() else {
306                break;
307            };
308            let span_start = span.start.saturating_sub(line_offset);
309            if span_start >= line_len {
310                break;
311            }
312            let span_end = cmp::min(span.start + span.len - line_offset, line_len);
313
314            if span_start > line_pos {
315                let spaces = " ".repeat(line[line_pos..span_start].width());
316                write!(out, "{spaces}")?;
317            }
318
319            let ch = span.kind.underline_char();
320            let underline: String =
321                iter::repeat_n(ch, line[span_start..span_end].width()).collect();
322            out.set_color(&span.kind.highlight_spec())?;
323            write!(out, "{underline}")?;
324            out.reset()?;
325
326            line_pos = span_end;
327            if span.start + span.len <= line_offset + line_len {
328                // Span is finished on this line; can proceed to the next one.
329                spans_iter.next();
330            }
331        }
332        writeln!(out)
333    }

335    pub fn write_as_table(&self, out: &mut impl WriteColor) -> io::Result<()> {
336        const POS_WIDTH: usize = 10;
337        const STYLE_WIDTH: usize = 22; // `buid magenta*/magenta*`
338
339        // Write table header.
340        let mut table_header_spec = ColorSpec::new();
341        table_header_spec.set_bold(true);
342        table_header_spec.set_intense(true);
343        out.set_color(&table_header_spec)?;
344        writeln!(
345            out,
346            "{pos:^POS_WIDTH$} {lhs:^STYLE_WIDTH$} {rhs:^STYLE_WIDTH$}",
347            pos = "Positions",
348            lhs = "Expected style",
349            rhs = "Actual style"
350        )?;
351        writeln!(
352            out,
353            "{pos:=>POS_WIDTH$} {lhs:=>STYLE_WIDTH$} {rhs:=>STYLE_WIDTH$}",
354            pos = "",
355            lhs = "",
356            rhs = ""
357        )?;
358        out.reset()?;
359
360        // Write table itself.
361        for differing_span in &self.differing_spans {
362            let start = differing_span.start;
363            let end = start + differing_span.len;
364            let pos = format!("{start}..{end}");
365            write!(out, "{pos:>POS_WIDTH$} ")?;
366
367            Self::write_color_spec(out, &differing_span.lhs_color_spec)?;
368            out.write_all(b" ")?;
369            Self::write_color_spec(out, &differing_span.rhs_color_spec)?;
370            writeln!(out)?;
371        }
372
373        Ok(())
374    }

376    /// Writes `color_spec` in human-readable format.
377    fn write_color_spec(out: &mut impl WriteColor, color_spec: &ColorSpec) -> io::Result<()> {
378        const COLOR_WIDTH: usize = 8; // `magenta*` is the widest color output
379
380        out.set_color(color_spec)?;
381        out.write_all(if color_spec.bold() { b"b" } else { b"-" })?;
382        out.write_all(if color_spec.italic() { b"i" } else { b"-" })?;
383        out.write_all(if color_spec.underline() { b"u" } else { b"-" })?;
384        out.write_all(if color_spec.dimmed() { b"d" } else { b"-" })?;
385
386        write!(
387            out,
388            " {fg:>COLOR_WIDTH$}/{bg:<COLOR_WIDTH$}",
389            fg = color_spec
390                .fg()
391                .map_or_else(|| "(none)".to_owned(), |&fg| Self::color_to_string(fg)),
392            bg = color_spec
393                .bg()
394                .map_or_else(|| "(none)".to_owned(), |&bg| Self::color_to_string(bg)),
395        )?;
396        out.reset()
397    }

399    fn color_to_string(color: Color) -> String {
400        match color {
401            Color::Black | Color::Ansi256(0) => "black".to_owned(),
402            Color::Red | Color::Ansi256(1) => "red".to_owned(),
403            Color::Green | Color::Ansi256(2) => "green".to_owned(),
404            Color::Yellow | Color::Ansi256(3) => "yellow".to_owned(),
405            Color::Blue | Color::Ansi256(4) => "blue".to_owned(),
406            Color::Magenta | Color::Ansi256(5) => "magenta".to_owned(),
407            Color::Cyan | Color::Ansi256(6) => "cyan".to_owned(),
408            Color::White | Color::Ansi256(7) => "white".to_owned(),
409
410            Color::Ansi256(8) => "black*".to_owned(),
411            Color::Ansi256(9) => "red*".to_owned(),
412            Color::Ansi256(10) => "green*".to_owned(),
413            Color::Ansi256(11) => "yellow*".to_owned(),
414            Color::Ansi256(12) => "blue*".to_owned(),
415            Color::Ansi256(13) => "magenta*".to_owned(),
416            Color::Ansi256(14) => "cyan*".to_owned(),
417            Color::Ansi256(15) => "white*".to_owned(),
418
419            Color::Rgb(r, g, b) => format!("#{r:02x}{g:02x}{b:02x}"),
420
421            _ => unreachable!(), // must be transformed during color normalization
422        }
423    }

426#[derive(Debug, Clone, Copy)]
427enum SpanHighlightKind {
428    Main,
429    Aux,
430}
431
432impl SpanHighlightKind {
433    fn underline_char(self) -> char {
434        match self {
435            Self::Main => '^',
436            Self::Aux => '!',
437        }
438    }
439
440    fn highlight_spec(self) -> ColorSpec {
441        let mut spec = ColorSpec::new();
442        match self {
443            Self::Main => {
444                spec.set_fg(Some(Color::White)).set_bg(Some(Color::Red));
445            }
446            Self::Aux => {
447                spec.set_fg(Some(Color::Black)).set_bg(Some(Color::Yellow));
448            }
449        }
450        spec
451    }
452}

454#[derive(Debug, Clone, Copy)]
455struct HighlightedSpan {
456    start: usize,
457    len: usize,
458    kind: SpanHighlightKind,
459}
460
461impl HighlightedSpan {
462    fn new(differing_spans: &[DiffColorSpan]) -> Vec<Self> {
463        let mut sequential_span_count = 1;
464        let span_highlights = differing_spans.windows(2).map(|window| match window {
465            [prev, next] => {
466                if prev.start + prev.len == next.start {
467                    sequential_span_count += 1;
468                } else {
469                    sequential_span_count = 1;
470                }
471                if sequential_span_count % 2 == 0 {
472                    SpanHighlightKind::Aux
473                } else {
474                    SpanHighlightKind::Main
475                }
476            }
477            _ => unreachable!(),
478        });
479
480        iter::once(SpanHighlightKind::Main)
481            .chain(span_highlights)
482            .zip(differing_spans)
483            .map(|(kind, span)| Self {
484                start: span.start,
485                len: span.len,
486                kind,
487            })
488            .collect()
489    }
