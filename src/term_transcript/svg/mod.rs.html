1//! Provides templating logic for rendering terminal output in a visual format.
2//!
3//! The included templating logic allows rendering SVG images. Templating is based on [Handlebars],
4//! and can be [customized](Template#customization) to support differing layout or even
5//! data formats (e.g., HTML). The default template supports [a variety of options](TemplateOptions)
6//! controlling output aspects, e.g. image dimensions and scrolling animation.
7//!
8//! [Handlebars]: https://handlebarsjs.com/
9//!
10//! # Examples
11//!
12//! See [`Template`] for examples of usage.

77/// Configurable options of a [`Template`].
78///
79/// # Serialization
80///
81/// Options can be deserialized from `serde`-supported encoding formats, such as TOML. This is used
82/// in the CLI app to read options from a file:
83///
84/// ```
85/// # use assert_matches::assert_matches;
86/// # use term_transcript::svg::{RgbColor, TemplateOptions, WrapOptions};
87/// let options_toml = r#"
88/// width = 900
89/// window_frame = true
90/// line_numbers = 'continuous'
91/// wrap.hard_break_at = 100
92/// scroll = { max_height = 300, pixels_per_scroll = 18, interval = 1.5 }
93///
94/// [palette.colors]
95/// black = '#3c3836'
96/// red = '#b85651'
97/// green = '#8f9a52'
98/// yellow = '#c18f41'
99/// blue = '#68948a'
100/// magenta = '#ab6c7d'
101/// cyan = '#72966c'
102/// white = '#a89984'
103///
104/// [palette.intense_colors]
105/// black = '#5a524c'
106/// red = '#b85651'
107/// green = '#a9b665'
108/// yellow = '#d8a657'
109/// blue = '#7daea3'
110/// magenta = '#d3869b'
111/// cyan = '#89b482'
112/// white = '#ddc7a1'
113/// "#;
114///
115/// let options: TemplateOptions = toml::from_str(options_toml)?;
116/// assert_eq!(options.width, 900);
117/// assert_matches!(options.wrap, Some(WrapOptions::HardBreakAt(100)));
118/// assert_eq!(
119///     options.palette.colors.green,
120///     RgbColor(0x8f, 0x9a, 0x52)
121/// );
122/// # anyhow::Ok(())
123/// ```
124#[derive(Debug, Serialize, Deserialize)]
125pub struct TemplateOptions {
126    /// Width of the rendered terminal window in pixels. Excludes the line numbers width if line
127    /// numbering is enabled. The default value is `720`.
128    #[serde(default = "TemplateOptions::default_width")]
129    pub width: usize,
130    /// Line height relative to the font size. If not specified, will be taken from font metrics (if a font is embedded),
131    /// or set to 1.2 otherwise.
132    pub line_height: Option<f64>,
133    /// Advance width of a font relative to the font size (i.e., in em units). If not specified, will be taken from font metrics (if a font is embedded),
134    /// or set to 8px (~0.57em) otherwise.
135    ///
136    /// For now, advance width is only applied to the pure SVG template.
137    // FIXME: extract to pure SVG options?
138    pub advance_width: Option<f64>,
139    /// Palette of terminal colors. The default value of [`Palette`] is used by default.
140    #[serde(default)]
141    pub palette: Palette,
142    /// CSS instructions to add at the beginning of the SVG `<style>` tag. This is mostly useful
143    /// to import fonts in conjunction with `font_family`.
144    ///
145    /// The value is not validated in any way, so supplying invalid CSS instructions can lead
146    /// to broken SVG rendering.
147    #[serde(skip_serializing_if = "str::is_empty", default)]
148    pub additional_styles: String,
149    /// Font family specification in the CSS format. Should be monospace.
150    #[serde(default = "TemplateOptions::default_font_family")]
151    pub font_family: String,
152    /// Indicates whether to display a window frame around the shell. Default value is `false`.
153    #[serde(default)]
154    pub window_frame: bool,
155    /// Options for the scroll animation. If set to `None` (which is the default),
156    /// no scrolling will be enabled, and the height of the generated image is not limited.
157    #[serde(skip_serializing_if = "Option::is_none", default)]
158    pub scroll: Option<ScrollOptions>,
159    /// Text wrapping options. The default value of [`WrapOptions`] is used by default.
160    #[serde(default = "TemplateOptions::default_wrap")]
161    pub wrap: Option<WrapOptions>,
162    /// Line numbering options.
163    #[serde(default)]
164    pub line_numbers: Option<LineNumbers>,
165    /// *Font embedder* that will embed the font into the SVG file via `@font-face` CSS.
166    /// This guarantees that the SVG will look identical on all platforms.
167    #[serde(skip)]
168    pub font_embedder: Option<Box<dyn FontEmbedder<Error = BoxedError>>>,
169}

217    /// Generates data for rendering.
218    ///
219    /// # Errors
220    ///
221    /// Returns an error if output cannot be rendered to HTML (e.g., it contains invalid
222    /// SGR sequences).
223    #[cfg_attr(
224        feature = "tracing",
225        tracing::instrument(level = "debug", skip(transcript), err)
226    )]
227    pub fn render_data<'s>(
228        &'s self,
229        transcript: &'s Transcript,
230    ) -> Result<HandlebarsData<'s>, TermError> {
231        let rendered_outputs = self.render_outputs(transcript)?;
232        let mut has_failures = false;
233
234        let mut used_chars = BTreeSet::new();
235        for interaction in transcript.interactions() {
236            let output = interaction.output().to_plaintext()?;
237            used_chars.extend(output.chars());
238
239            let input = interaction.input();
240            if !input.hidden {
241                let prompt = input.prompt.as_deref();
242                let input_chars = iter::once(input.text.as_str())
243                    .chain(prompt)
244                    .flat_map(str::chars);
245                used_chars.extend(input_chars);
246            }
247        }
248        if self.line_numbers.is_some() {
249            used_chars.extend('0'..='9');
250        }
251        if self.wrap.is_some() {
252            used_chars.insert('»');
253        }
254
255        let embedded_font = self
256            .font_embedder
257            .as_deref()
258            .map(|embedder| embedder.embed_font(used_chars))
259            .transpose()
260            .map_err(TermError::FontEmbedding)?;
261
262        let interactions: Vec<_> = transcript
263            .interactions()
264            .iter()
265            .zip(rendered_outputs)
266            .map(|(interaction, output)| {
267                let failure = interaction
268                    .exit_status()
269                    .is_some_and(|status| !status.is_success());
270                has_failures = has_failures || failure;
271                SerializedInteraction {
272                    input: interaction.input(),
273                    output,
274                    exit_status: interaction.exit_status().map(|status| status.0),
275                    failure,
276                }
277            })
278            .collect();
279
280        Ok(HandlebarsData {
281            creator: CreatorData::default(),
282            interactions,
283            options: self,
284            has_failures,
285            embedded_font,
286        })
287    }
288
289    #[cfg_attr(
290        feature = "tracing",
291        tracing::instrument(level = "debug", skip_all, err)
292    )]
293    fn render_outputs(&self, transcript: &Transcript) -> Result<Vec<Vec<StyledLine>>, TermError> {
294        let max_width = self.wrap.as_ref().map(|wrap_options| match wrap_options {
295            WrapOptions::HardBreakAt(width) => *width,
296        });
297
298        transcript
299            .interactions
300            .iter()
301            .map(|interaction| interaction.output().to_lines(max_width))
302            .collect()
303    }
304}

306/// Options that influence the scrolling animation.
307///
308/// The animation is only displayed if the console exceeds [`Self::max_height`]. In this case,
309/// the console will be scrolled vertically by [`Self::pixels_per_scroll`]
310/// with the interval of [`Self::interval`] seconds between every frame.
311#[derive(Debug, Clone, Deserialize, Serialize)]
312pub struct ScrollOptions {
313    /// Maximum height of the console, in pixels. The default value allows to fit 19 lines
314    /// of text into the view with the default template (potentially, slightly less because
315    /// of vertical margins around user inputs).
316    pub max_height: usize,
317    /// Number of pixels moved each scroll. Default value is 52 (~3 lines of text with the default template).
318    pub pixels_per_scroll: usize,
319    /// Interval between keyframes in seconds. The default value is `4`.
320    pub interval: f32,
321}
322
323impl Default for ScrollOptions {
324    fn default() -> Self {
325        const DEFAULT_LINE_HEIGHT: usize = 18; // from the default template
326        Self {
327            max_height: DEFAULT_LINE_HEIGHT * 19,
328            pixels_per_scroll: 52,
329            interval: 4.0,
330        }
331    }
332}

350/// Template for rendering [`Transcript`]s, e.g. into an [SVG] image.
351///
352/// # Available templates
353///
354/// When using a template created with [`Self::new()`], a transcript is rendered into SVG
355/// with the text content embedded as an HTML fragment. This is because SVG is not good
356/// at laying out multiline texts and text backgrounds, while HTML excels at both.
357/// As a downside of this approach, the resulting SVG requires for its viewer to support
358/// HTML embedding; while web browsers *a priori* support such embedding, some other SVG viewers
359/// may not.
360///
361/// A template created with [`Self::pure_svg()`] renders a transcript into pure SVG,
362/// in which text is laid out manually and backgrounds use a hack (lines of text with
363/// appropriately colored `█` chars placed behind the content lines). The resulting SVG is
364/// supported by more viewers, but it may look incorrectly in certain corner cases. For example,
365/// if the font family used in the template does not contain `█` or some chars
366/// used in the transcript, the background may be mispositioned.
367///
368/// [Snapshot testing](crate::test) functionality produces snapshots using [`Self::new()`]
369/// (i.e., with HTML embedding); pure SVG templates cannot be tested.
370///
371/// # Customization
372///
373/// A custom [Handlebars] template can be supplied via [`Self::custom()`]. This can be used
374/// to partially or completely change rendering logic, including the output format (e.g.,
375/// to render to HTML instead of SVG).
376///
377/// Data supplied to a template is [`HandlebarsData`].
378///
379/// Besides [built-in Handlebars helpers][rust-helpers] (a superset of [standard helpers]),
380/// custom templates have access to the following additional helpers. All the helpers are
381/// extensively used by the [default template]; thus, studying it may be a good place to start
382/// customizing. Another example is an [HTML template] from the crate examples.
383///
384/// ## Arithmetic helpers: `add`, `sub`, `mul`, `div`
385///
386/// Perform the specified arithmetic operation on the supplied args.
387/// `add` and `mul` support any number of numeric args; `sub` and `div` exactly 2 numeric args.
388/// `div` also supports rounding via `round` hash option. `round=true` rounds to the nearest
389/// integer; `round="up"` / `round="down"` perform rounding in the specified direction.
390///
391/// ```handlebars
392/// {{add 2 3 5}}
393/// {{div (len xs) 3 round="up"}}
394/// ```
395///
396/// ## Counting lines: `count_lines`
397///
398/// Counts the number of lines in the supplied string. If `format="html"` hash option is included,
399/// line breaks introduced by `<br/>` tags are also counted.
400///
401/// ```handlebars
402/// {{count_lines test}}
403/// {{count_lines test format="html"}}
404/// ```
405///
406/// ## Integer ranges: `range`
407///
408/// Creates an array with integers in the range specified by the 2 provided integer args.
409/// The "from" bound is inclusive, the "to" one is exclusive.
410///
411/// ```handlebars
412/// {{#each (range 0 3)}}{{@index}}, {{/each}}
413/// {{! Will output `0, 1, 2,` }}
414/// ```
415///
416/// ## Variable scope: `scope`
417///
418/// A block helper that creates a scope with variables specified in the options hash.
419/// In the block, each variable can be obtained or set using an eponymous helper:
420///
421/// - If the variable helper is called as a block helper, the variable is set to the contents
422///   of the block, which is treated as JSON.
423/// - If the variable helper is called as an inline helper with the `set` option, the variable
424///   is set to the value of the option.
425/// - Otherwise, the variable helper acts as a getter for the current value of the variable.
426///
427/// ```handlebars
428/// {{#scope test=""}}
429///   {{test set="Hello"}}
430///   {{test}} {{! Outputs `Hello` }}
431///   {{#test}}{{test}}, world!{{/test}}
432///   {{test}} {{! Outputs `Hello, world!` }}
433/// {{/scope}}
434/// ```
435///
436/// Since variable getters are helpers, not "real" variables, they should be enclosed
437/// in parentheses `()` if used as args / options for other helpers, e.g. `{{add (test) 2}}`.
438///
439/// ## Partial evaluation: `eval`
440///
441/// Evaluates a partial with the provided name and parses its output as JSON. This can be used
442/// to define "functions" for better code structuring. Function args can be supplied in options
443/// hash.
444///
445/// ```handlebars
446/// {{#*inline "some_function"}}
447///   {{add x y}}
448/// {{/inline}}
449/// {{#with (eval "some_function" x=3 y=5) as |sum|}}
450///   {{sum}} {{! Outputs 8 }}
451/// {{/with}}
452/// ```
453///
454/// [SVG]: https://developer.mozilla.org/en-US/docs/Web/SVG
455/// [Handlebars]: https://handlebarsjs.com/
456/// [rust-helpers]: https://docs.rs/handlebars/latest/handlebars/index.html#built-in-helpers
457/// [standard helpers]: https://handlebarsjs.com/guide/builtin-helpers.html
458/// [default template]: https://github.com/slowli/term-transcript/blob/master/src/svg/default.svg.handlebars
459/// [HTML template]: https://github.com/slowli/term-transcript/blob/master/examples/custom.html.handlebars
460///
461/// # Examples
462///
463/// ```
464/// use term_transcript::{svg::*, Transcript, UserInput};
465///
466/// # fn main() -> anyhow::Result<()> {
467/// let mut transcript = Transcript::new();
468/// transcript.add_interaction(
469///     UserInput::command("test"),
470///     "Hello, \u{1b}[32mworld\u{1b}[0m!",
471/// );
472///
473/// let template_options = TemplateOptions {
474///     palette: NamedPalette::Dracula.into(),
475///     ..TemplateOptions::default()
476/// };
477/// let mut buffer = vec![];
478/// Template::new(template_options).render(&transcript, &mut buffer)?;
479/// let buffer = String::from_utf8(buffer)?;
480/// assert!(buffer.contains(r#"Hello, <span class="fg2">world</span>!"#));
481/// # Ok(())
482/// # }
483/// ```
484pub struct Template {
485    options: TemplateOptions,
486    handlebars: Handlebars<'static>,
487    constants: HashMap<&'static str, u32>,
488}
489

541    /// Initializes a custom template.
542    #[allow(clippy::missing_panics_doc)] // Panic should never be triggered
543    pub fn custom(template: HandlebarsTemplate, options: TemplateOptions) -> Self {
544        let mut handlebars = Handlebars::new();
545        handlebars.set_strict_mode(true);
546        register_helpers(&mut handlebars);
547        handlebars.register_template(MAIN_TEMPLATE_NAME, template);
548        let helpers = HandlebarsTemplate::compile(COMMON_HELPERS).unwrap();
549        handlebars.register_template("_helpers", helpers);
550        Self {
551            options,
552            handlebars,
553            constants: HashMap::new(),
554        }
555    }

557    /// Renders the `transcript` using the template (usually as an SVG image, although
558    /// custom templates may use a different output format).
559    ///
560    /// # Errors
561    ///
562    /// Returns a Handlebars rendering error, if any. Normally, the only errors could be
563    /// related to I/O (e.g., the output cannot be written to a file).
564    #[cfg_attr(feature = "tracing", tracing::instrument(skip_all, err))]
565    pub fn render<W: Write>(
566        &self,
567        transcript: &Transcript,
568        destination: W,
569    ) -> Result<(), RenderError> {
570        let data = self
571            .options
572            .render_data(transcript)
573            .map_err(|err| RenderErrorReason::NestedError(Box::new(err)))?;
574        let data = CompleteHandlebarsData {
575            inner: data,
576            constants: &self.constants,
577        };
578        #[cfg(feature = "tracing")]
579        tracing::debug!(?data, "using Handlebars data");
580
581        #[cfg(feature = "tracing")]
582        let _entered = tracing::debug_span!("render_to_write").entered();
583        self.handlebars
584            .render_to_write(MAIN_TEMPLATE_NAME, &data, destination)
585    }
